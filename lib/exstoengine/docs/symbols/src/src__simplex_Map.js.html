1 ex.using([ 
  2     'ex.simplex.Layer' 
  3 ], function() {
  4 	ex.define("ex.simplex.Map", {
  5 		/**
  6 		 * Builds the map with a name and an empty set of layers
  7 		 * 
  8 		 * @name ex.simplex.Map
  9 		 * 
 10 		 * @param {String} name name of map
 11 		 * 
 12 		 * @property {String} name
 13 		 * @property {ex.simplex.Layer[]} layers
 14 		 * 
 15 		 * @constructor
 16 		 */
 17 		constructor : function(name) {
 18 			this.name = name;
 19 			this.layers = [];
 20 		},
 21 
 22 		/**
 23 		 * Adds a layer to the the end of layers.
 24 		 * 
 25 		 * @function
 26 		 * @name addLayer
 27 		 * @memberOf ex.simplex.Map
 28 		 * 
 29 		 * @param layer
 30 		 */
 31 		addLayer : function(layer) {
 32 			if (layer != null){
 33 				this.layers.push(layer);
 34 			}
 35 		},
 36 		
 37 		/**
 38 		 * Retrieves a layer from the map by name.
 39 		 * 
 40 		 * @function
 41 		 * @name getLayer
 42 		 * @memberOf ex.simplex.Map
 43 		 * 
 44 		 * @param {String} name
 45 		 * @returns {ex.simplex.Layer} returns null if no layer is found
 46 		 * 		by the supplied name.
 47 		 */
 48 		getLayer: function(name) {
 49 			if(name == null)
 50 				return null;
 51 			
 52 			var index = this.layers.length;
 53 			while(index--){
 54 				if(this.layers[index].name == name){
 55 					return this.layers[index];
 56 				}
 57 			}
 58 			
 59 			// if not found...
 60 			return null;
 61 		},
 62 
 63 		/**
 64 		 * Removes the layer at the specified index.
 65 		 * 
 66 		 * @function
 67 		 * @name removeLayer
 68 		 * @memberOf ex.simplex.Map
 69 		 * 
 70 		 * @param {Number} index location of layer in array
 71 		 */
 72 		removeLayer : function(index) {
 73 			if (index < 0 || index >= this.layers.length) {
 74 				return;
 75 			}
 76 			this.layers.splice(index, 1);
 77 		},
 78 
 79 		/**
 80 		 * Toggles layer visiblity.
 81 		 * 
 82 		 * @function
 83 		 * @name toggleLayer
 84 		 * @memberOf ex.simplex.Map
 85 		 * 
 86 		 * @param {Number} layerId index of layer to toggle
 87 		 */
 88 		toggleLayer : function(layerId) {
 89 			var layer = this.layers[layerId];
 90 			layer.visible = !layer.visible;
 91 		},
 92 
 93 		/**
 94 		 * Performs actions every time period dt.
 95 		 * 
 96 		 * @function
 97 		 * @name update
 98 		 * @memberOf ex.simplex.Map
 99 		 * 
100 		 * @param {Number} dt delta time, length of each time cycle
101 		 */
102 		update : function(dt) {
103 			var index = this.layers.length;
104 			while(index--){
105 				this.layers[index].update(dt);
106 			}
107 		},
108 
109 		/**
110 		 * Supplies a canvas context and camera offset to each item and calls
111 		 * their render functions
112 		 * 
113 		 * @function
114 		 * @name render
115 		 * @memberOf ex.simplex.Map
116 		 * 
117 		 * @param {Context} context canvas context to draw with
118 		 * @param {Number} camX camera offset on x
119 		 * @param {Number} camY camera offset on y
120 		 */
121 		render : function($context, $camX, $camY) {
122 			var i = this.layers.length;
123 			while (i--) {
124 				this.layers[i].render($context, $camX, $camY);
125 			}
126 		}
127 	});
128 });
