1 /**
  2  * 
  3  * @class ex.simplex.Entity
  4  * @param name
  5  *            {String}: name of entity
  6  * @param position
  7  *            {ex.base.Point}: position of entity
  8  * @param actions
  9  *            {Object}: contains properties that perform actions
 10  * @param bindings
 11  *            {Object}: associative array that binds inputs to actions
 12  */
 13 ex.using([ 
 14     "ex.event.EventTarget"
 15 ], function() {
 16 	ex.define("ex.simplex.Entity", ex.event.EventTarget, {
 17 
 18 		/**
 19 		 * The base class for all interactive game objects such as players,
 20 		 * NPCs, and triggers.
 21 		 * 
 22 		 * @name ex.world.Entity
 23 		 * 
 24 		 * @param {String} name name of entity (used mostly in IDE)
 25 		 * @param {ex.base.Vector} position original position of entity
 26 		 * @param {ex.base.Sprite} sprite display object
 27 		 * @param {Boolean} collides whether collision should be added
 28 		 * 		to the entity.
 29 		 * 
 30 		 * @property {String} name
 31 		 * @property {String} type
 32 		 * 		Internal use only! Do not change!
 33 		 * @property {ex.base.Vector} position
 34 		 * @property {ex.base.Vector} velocity
 35 		 * @property {Number} width
 36 		 * @property {Number} height
 37 		 * @property {Boolean} collides
 38 		 * @property {Number} mass used in physics engine
 39 		 * @property {ex.display.Sprite} sprite
 40 		 * @constructor
 41 		 */
 42 		constructor : function(name, position, sprite, collides) {
 43 			// Referencing data
 44 			this.name = name;
 45 			this.type = "Entity";
 46 			// Physical data
 47 			this.position = position;
 48 			this.width = sprite.width;
 49 			this.height = sprite.height;
 50 			this.velocity = new ex.base.Vector(0,0);
 51 			this.collides = collides;
 52 			this.mass = 1;
 53 			// Display data
 54 			this.sprite = sprite;
 55 			this.sprite.position = this.position;	// pointer to this.position
 56 			
 57 			if(sprite.width == 0 && sprite.height == 0) {
 58 				ex.event.listen(sprite.img, 'load', function () {
 59 					this.width = sprite.width;
 60 					this.height = sprite.height;
 61             	}, this);
 62 			}
 63 		},
 64 
 65 		/**
 66 		 * performs actions every time period dt
 67 		 * 
 68 		 * @function
 69 		 * @name update
 70 		 * @memberOf ex.simplex.Entity
 71 		 * 
 72 		 * @param {Number} dt delta time, length of each time cycle
 73 		 */
 74 		update : function(dt) {
 75 			this.sprite.update(dt);
 76 			this.updatePosition(this.velocity, dt);
 77 		},
 78 		
 79 		/**
 80 		 * Adds a time scaled vector to the entity's position.
 81 		 * This must be used to update the entity's position or the entity will
 82 		 * become out of sync with its sprite.
 83 		 * 
 84 		 * If you want to set the absolute position, see setPosition(newX, newY)
 85 		 * @see ex.simplex.Entity.setPosition
 86 		 * 
 87 		 * @function
 88 		 * @name updatePosition
 89 		 * @memberOf ex.simplex.Entity
 90 		 * 
 91 		 * @param {ex.base.Vector} vector
 92 		 * @param {Number} dt
 93 		 */
 94 		updatePosition: function(vector, dt) {
 95 			this.position.addScaled(vector, dt);
 96 		},
 97 		
 98 		/**
 99 		 * Sets the absolute position of the entity with x,y values.
100 		 * 
101 		 * @function
102 		 * @name setPosition
103 		 * @memberOf ex.simplex.Entity
104 		 * 
105 		 * @param {Number} newX
106 		 * @param {Number} newY
107 		 */
108 		setPosition: function(newX, newY) {
109 			this.position.x = newX;
110 			this.position.y = newY;
111 		},
112 		
113 		/**
114 		 * Called every time a collision is found with the entity.
115 		 * 
116 		 * @function
117 		 * @name onCollide
118 		 * @memberOf ex.simplex.Entity
119 		 * 
120 		 * @param {Entity} target the collision target
121 		 */
122 		onCollide: function(target) {
123 			
124 		},
125 
126 		/**
127 		 * Checks properties and determines if entity is visible.
128 		 * 
129 		 * @function
130 		 * @name isVisible
131 		 * @memberOf ex.simplex.Entity
132 		 * 
133 		 * @returns {Boolean}
134 		 */
135 		isVisible : function() {
136 			return this.visible;
137 		},
138 
139 		/**
140 		 * Supplies a canvas context and camera offset to each item and calls
141 		 * their render functions
142 		 * 
143 		 * @param {Context} context canvas context to draw with
144 		 * @param {Number} camX camera offset on x
145 		 * @param {Number} camY camera offset on y
146 		 */
147 		render : function(context, camX, camY) {
148 			if (!this.isVisible()) // Don't render if it won't be seen
149 				return;
150 
151 			// Render code
152 		}
153 	});
154 });
