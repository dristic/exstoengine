1 ex.using([
  2   "ex.world.TileMap",
  3   "ex.display.Renderable"
  4 ], function () {
  5 	ex.define("ex.display.SpriteMap", ex.display.Renderable, {
  6 		
  7 		/**
  8 		 * The graphical representation of a TileMap.
  9 		 * 
 10 		 * @name ex.display.SpriteMap
 11 		 * @extends ex.world.TileMap
 12 		 * 
 13 		 * @param {Number} tileWidth
 14 		 * @param {Number} tileHeight
 15 		 * @param {Number[]} map a 2D array of integer values relating to the
 16 		 * 		tiles in the supplied tileSet
 17 		 * @param {Image} tileSet
 18 		 * @param {String} name
 19 		 * 
 20 		 * @property {Boolean} collides Controls whether the object is included
 21 		 * 		in collision detection.
 22 		 * @propety {String} name
 23 		 * @property {Image} tileSet the tile set to use when rendering.
 24 		 * @property {ex.base.Vector} position
 25 		 * @property {ex.base.Vector} scrollFactor
 26 		 */
 27 		constructor: function(tileWidth, tileHeight, map, tileSet, name, yOffset) {
 28 			this.collides = false;
 29 			this.name = name;
 30 			this.tileSet = tileSet;
 31 			this.position = new ex.base.Vector(0,0);
 32 			this.scrollFactor = new ex.base.Vector(1,1);
 33 			this.yOffset = yOffset || 0;
 34 
 35 			// Call Renderable constructor (visibility, opacity)
 36 			this._super("constructor", [true, 1.0]);
 37 			
 38 			// Create TileMap
 39 			this.tileMap = new ex.world.TileMap(tileWidth, tileHeight, map);
 40 
 41 			// Used to pre-render the whole map as an image
 42 			this.preRenderCanvas = document.createElement("canvas");
 43 			this.preRenderCanvas.width = this.tileMap.width;
 44 			this.preRenderCanvas.height = this.tileMap.height;
 45 			this.preRenderContext = this.preRenderCanvas.getContext('2d');
 46 			
 47 			this._preRenderSpriteMap();
 48 			this.rendered = 10;
 49 		},
 50 		
 51 		/**
 52 		 * Contains any event handling when a collision occurs between this
 53 		 * and another object.
 54 		 * 
 55 		 * @function
 56 		 * @name onCollide
 57 		 * @memberOf ex.display.SpriteMap
 58 		 */
 59 		onCollide: function() {
 60 			
 61 		},
 62 		
 63 		/**
 64 		 * The update routine.
 65 		 * 
 66 		 * @function
 67 		 * @name update
 68 		 * @memberOf ex.display.SpriteMap
 69 		 * 
 70 		 * @param {Number} dt timestep
 71 		 */
 72 		update: function(dt) {
 73 			if(this.rendered-- > 0) {
 74 				this._preRenderSpriteMap();
 75 			}
 76 			// Check for any tile changes
 77 			// If changes found, 
 78 			//		alter the preRenderCanvas accordingly
 79 		},
 80 		
 81 		/*
 82 		 * Pre-renders the SpriteMap to the preRenderCanvas,
 83 		 * which acts as a buffer to the SpriteMap. This
 84 		 * results in the SpriteMap not having to be fully
 85 		 * generated each frame, instead redrawing the buffer
 86 		 * as a single image.
 87 		 */
 88 		_preRenderSpriteMap: function () {
 89 			var yPos = this.tileMap.data.length - 1,
 90 				xPos = 0;
 91 			for(yPos; yPos > -1; yPos--) {
 92 				for(xPos; xPos < this.tileMap.data[yPos].length; xPos++) {
 93 					var tile = this.tileMap.data[yPos][xPos], sx = 0, sy = 0;
 94 					var tileValue = tile.value;
 95 					if(tileValue != 0) {
 96 						while(--tileValue) {
 97 							sx += this.tileMap.tileWidth;
 98 							if(sx >= this.tileSet.width) {
 99 								sy += this.tileMap.tileHeight;
100 								sx = 0;
101 							}
102 						}
103 						this.preRenderContext.drawImage(
104 								this.tileSet,
105 								sx,
106 								sy,
107 								this.tileMap.tileWidth,
108 								this.tileMap.tileHeight,
109 								tile.position.x,
110 								tile.position.y - (this.yOffset * yPos),
111 								tile.width,
112 								tile.height);
113 					}
114 				}
115 				xPos = 0;
116 			}
117 		},
118 		
119 		/**
120 		 * The render routine.
121 		 * 
122 		 * @function
123 		 * @name render
124 		 * @memberOf ex.display.SpriteMap
125 		 * 
126 		 * @param {Context} context
127 		 * @param {Number} camX
128 		 * @param {Number} camY
129 		 * @param {Number} camWidth
130 		 * @param {Number} camHeight
131 		 */
132 		render: function(context, camX, camY, camWidth, camHeight) {
133 			// Do nothing if not visible
134 			if(!this.isVisible()){
135 				return;
136 			}
137 			
138 			// Position of the sprite in the viewport
139 			var viewPortX = ex.toInt(this.position.x - (camX * this.scrollFactor.x)),
140 				viewPortY = ex.toInt(this.position.y - (camY * this.scrollFactor.y));
141 						
142 			// Do nothing if sprite map is out of the viewport
143 			if((viewPortX + this.preRenderCanvas.width) < 0
144 					|| viewPortX > camWidth
145 					||(viewPortY + this.preRenderCanvas.height) < 0
146 					|| viewPortY > camHeight) {
147 				return;
148 			}
149 			
150 			// Calculate section of preRenderCanvas to draw
151 			var sourceX = -viewPortX,
152 				sourceY = -viewPortY,
153 				sourceWidth = camWidth - viewPortX,
154 				sourceHeight = camWidth - viewPortY;
155 			var destX = viewPortX,
156 				destY = viewPortY;
157 			
158 			// Constrain values to image size
159 			if(sourceX < 0) sourceX = 0;
160 			if(sourceY < 0) sourceY = 0;
161 			if((sourceX + sourceWidth) > this.preRenderCanvas.width) sourceWidth = this.preRenderCanvas.width - sourceX;
162 			if((sourceY + sourceHeight) > this.preRenderCanvas.height) sourceHeight = this.preRenderCanvas.height - sourceY;
163 			if(destX < 0) destX = 0;
164 			if(destY < 0) destY = 0;
165 			
166 			// Draw image
167 			context.drawImage(
168             		this.preRenderCanvas,
169             		sourceX,
170             		sourceY,
171             		sourceWidth,
172             		sourceHeight,
173             		destX, 
174             		destY,
175             		sourceWidth,	//destWidth and Height == sourceWidth and Height
176             		sourceHeight);
177 		}
178 	});
179 });
