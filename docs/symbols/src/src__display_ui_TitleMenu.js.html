1 ex.using([
  2   'ex.display.Sprite',
  3   'ex.base.Component'
  4 ], function() {	
  5 	ex.define("ex.display.ui.TitleMenu", ex.base.Component, {
  6 		/**
  7 		 * Creates a title menu with background, logo, selection options,
  8 		 * and input bindings.
  9 		 * @name ex.display.ui.TitleMenu
 10 		 * @param {Object} selections a selection object (text, action pair)
 11 		 * @param {Number} defaultSelection starting selection index (0 based)
 12 		 * @param {Image} bgImage image to use as the background
 13 		 * @param {Image} logoImage image to use as the logo
 14 		 * @param {ex.Input} input the input component of the engine.
 15 		 * 
 16 		 * @property {Object[]} selections The array of selections
 17 		 * 		available on the menu.
 18 		 * @property {Number} currentSelection The array index of
 19 		 * 		the current selection.
 20 		 * @property {ex.util.Input} listening input component.
 21 		 * @property {Object} settings for orientation and selection events.
 22 		 * 
 23 		 * @constructor
 24 		 */
 25 		constructor: function(
 26 				selections, defaultSelection, bgImage, logoImage, input){
 27 			this.selections = selections;
 28 			this.currentSelection = defaultSelection;
 29 			this.input = input;
 30 			
 31 			this.options = {
 32 				background: {
 33 					image: bgImage
 34 				},
 35 				logo: {
 36 					image: logoImage,
 37 					x: 300,
 38 					y: 100
 39 				},
 40 				menu: {
 41 					x: 400,
 42 					y: 300,
 43 					actionKey: ex.util.Key.Enter
 44 				},
 45 				selection: {
 46 					width: 150,
 47 					height: 70
 48 				},
 49 			};
 50 		},
 51 		
 52 		/**
 53 		 * Moves the selection up by one if possible.
 54 		 * @function
 55 		 * @name moveUpMenu
 56 		 * @memberOf ex.display.ui.TitleMenu
 57 		 */
 58 		moveUpMenu: function() {
 59 			if(this.currentSelection != 0){
 60 				this.currentSelection--;
 61 			}
 62 		},
 63 		
 64 		/**
 65 		 * Moves the selection down by one if possible.
 66 		 * @function
 67 		 * @name moveDownMenu
 68 		 * @memberOf ex.display.ui.TitleMenu
 69 		 */
 70 		moveDownMenu: function() {
 71 			if(this.currentSelection < (this.selections.length - 1)) {
 72 				this.currentSelection++;
 73 			}
 74 		},
 75 		
 76 		/**
 77 		 * The update loop where user input is checked
 78 		 * @function
 79 		 * @name update
 80 		 * @memberOf ex.display.ui.TitleMenu
 81 		 * @param {Number} dt timestep
 82 		 */
 83 		update: function(dt){
 84 			if(this.input.isKeyPressed(this.options.menu.actionKey)) {
 85 				this.selections[this.currentSelection].action();
 86 			}
 87 			if(this.input.isKeyPressed(ex.util.Key.Up)) {
 88 				this.moveUpMenu();
 89 			}
 90 			if(this.input.isKeyPressed(ex.util.Key.Down)) {
 91 				this.moveDownMenu();
 92 			}
 93 			
 94 			// Mouse events for mobile devices or people using mice
 95 			// should go here...
 96 			// 		Requirements:
 97 			//		 - bounding box
 98 			//		 - click event to run action
 99 			//		 - mouseover event to set currentSelection
100 			if(this.input.mouseDown){
101 				var index = this.selections.length;
102 				while(index--){
103 					if(this.input.mouseX > (this.options.menu.x - this.options.selection.width) &&
104 							this.input.mouseX < (this.options.menu.x + this.options.selection.width) &&
105 							this.input.mouseY > (this.options.menu.y + (this.options.selection.height * (index - 1))) &&
106 							this.input.mouseY < (this.options.menu.y + (this.options.selection.height * (index)))){
107 						this.selections[index].action();
108 					}
109 				}
110 			}
111 		},
112 		
113 		/**
114 		 * Renders the TitleMenu to the screen.
115 		 * @function
116 		 * @name render
117 		 * @memberOf ex.display.ui.TitleMenu
118 		 * @param {Canvas Context} context canvas context to render with
119 		 * @param {Number} camX camera offset on x
120 		 * @param {Number} camY camera offset on y
121 		 */
122 		render: function(context, camX, camY){
123 			if(this.options.background.image != null) {
124 				context.drawImage(
125 						this.options.background.image,
126 						0,0);
127 			}
128 			
129 			if(this.options.logo.image != null) {
130 				context.drawImage(
131 						this.options.logo.image,
132 						this.options.logo.x,
133 						this.options.logo.y);
134 			}
135 			
136 			context.font = "40pt Calibri";
137 			context.fillStyle = "#FF0000";
138 			context.textAlign = "center";
139 			
140 			var xPos = this.options.menu.x;
141 			var yPos = this.options.menu.y;
142 			var index = 0;
143 			for(index; index < this.selections.length; index++) {
144 				// Color current selection differently
145 				if(index == this.currentSelection){
146 					context.fillStyle = "#00FF00";
147 					context.fillText(this.selections[index].text, xPos, yPos);
148 					context.fillStyle = "#FF0000";
149 				} else {
150 					// Print non selected options in normal color
151 					context.fillText(this.selections[index].text, xPos, yPos);
152 				}
153 				yPos += this.options.selection.height;
154 			}
155 		}
156 	});
157 });
