1 ex.using([
  2   "ex.base.Vector"

  4 	function vary(n) {
  5 		return (Math.random() * n << 1) - n;
  6 	};
  7 	
  8 	function extend(object, extend) {
  9 		for(var param in extend) {
 10 			object[param] = extend[param];
 11 		}
 12 	}
 13 	
 14 	ex.define("ex.display.Particle", {
 15 		constructor: function(options) {
 16 			var defaults = {
 17 				position: new ex.base.Vector(0, 0),
 18 				velocity: new ex.base.Vector(0, 0),
 19 				xVariance: 0,
 20 				yVariance: 0,
 21 				direction: new ex.base.Vector(0, 0),
 22 				directionVariance: 0,
 23 				age: 0,
 24 				life: 0,
 25 				lifeVariance: 0,
 26 				size: 0,
 27 				sizeVariance: 0,
 28 				active: true,
 29 				alpha: 1,
 30 				color: "#cef"
 31 			};
 32 			
 33 			defaults.extend(options);
 34 			
 35 			defaults.direction = options.direction + vary(options.directionVariance);
 36 			defaults.position = options.position.clone().add(defaults.position);
 37 			defaults.velocity = options.velocity.clone().rotate(defaults.direction * Math.PI / 180);
 38 			defaults.life = options.life + vary(options.lifeVariance);

 40 			
 41 			this.extend(defaults);

 44 		update: function(dt) {
 45 			if(this.age >= this.life) this.active = false;

 47 			this.alpha -= 0.1;
 48 			this.age++;
 49 		},
 50

 53 			
 54 			context.save();
 55 			
 56 			context.fillStyle = this.color;
 57 			try {
 58 				context.globalAlpha = this.alpha;
 59 			} catch(e) {
 60 				
 61 			}
 62 			context.translate(this.position.x - (camX * this.scrollFactorX), this.position.y - (camY * this.scrollFactorY));
 63 			
 64 			context.beginPath();
 65 	        context.arc(0, 0, this.size/2, 0, Math.PI/180, true);
 66 	        context.closePath();

 69 	        context.restore();
 70 		}
