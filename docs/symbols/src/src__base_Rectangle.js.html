1 ex.using([
  2     "ex.base.Point"          
  3 ],function () {
  4 	ex.define("ex.base.Rectangle", {
  5 		/**
  6 		 * @name ex.base.Rectangle
  7 		 * @param {ex.base.Point} position of top left corner
  8 		 * @param {Number} width width of rectangle
  9 		 * @param {Number} height height of rectangle
 10 		 * 
 11 		 * @property {ex.base.Point} position top left corner
 12 		 * @property {Number} width
 13 		 * @property {Number} height
 14 		 * @constructor
 15 		 */
 16 		constructor: function(position, width, height) {
 17 			this.position = position;
 18 			this.width = width;
 19 			this.height = height;
 20 		},
 21 		
 22 		/**
 23 		 * Checks if the supplied point is contained in the Rectangle
 24 		 * @function
 25 		 * @name containsPoint
 26 		 * @memberOf ex.base.Rectangle
 27 		 * @param {Number or ex.base.Point} x 
 28 		 * 			point's x axis value or an ex.base.Point
 29 		 * @param {Number} [y] 
 30 		 * 			point's y axis value (not used if x is an ex.base.Point)
 31 		 * @returns {Boolean}
 32 		 */
 33 		containsPoint: function(x, y) {
 34 			if(x instanceof ExstoEngine.Base.Point) {
 35 				if(x.x > this.x && x.x  < this.position.x + this.width 
 36 				   && x.y > this.y && x.y < this.position.y + this.height) {
 37 					return true;
 38 				} else {
 39 					return false;
 40 				}
 41 			} else {
 42 				if(x > this.position.x && x < this.position.x + this.width
 43 				   && y > this.position.y && y < this.position.y + this.height) {
 44 			   		return true;
 45 			   } else {
 46 			   		return false;
 47 			   }
 48 			}
 49 		},
 50 		
 51 		/**
 52 		 * Checks if the supplied point is contained in the Rectangle
 53 		 * @function
 54 		 * @name translate
 55 		 * @memberOf ex.base.Rectangle
 56 		 * @param {Number, ex.base.Point, or ex.base.Vector} x 
 57 		 * 			translation on x axis value, ex.base.Point offset, 
 58 		 * or ex.base.Vector offset
 59 		 * @param {Number} [y] 
 60 		 * 			translation on y axis value (not used if x is an 
 61 		 * ex.base.Point or ex.base.Vector)
 62 		 * @return {ex.base.Rectangle} this
 63 		 */
 64 		translate: function(x, y) {
 65 			if(x instanceof ex.base.Point || x instanceof ex.base.Vector){
 66 				this.position.x -= x.x;
 67 				this.position.y -= x.y;
 68 			} else {
 69 				this.position.x -= x;
 70 				this.position.y -= y;
 71 			}
 72 			return this;
 73 		}
 74 	});
 75 });
