 10 /* Requirements:
 11  * - Support for animated sprites as particles
 12  * - drawing objects as particles
 13  * - variance in all directions (angle or cartesian)
 14  * - variance in age/size
 15  * - spawn rate
 16  * - initial velocity
 17  * - degree of randomness
 18  * - simulation and render phases
 19  * - emitters on points, lines, and maybe polys? (lines could be used for rain effects)
 20  */
 21 
 22 ex.using([
 23   "ex.base.Vector",
 24   "ex.plugins.Particle2"
 25 ], function () {
 26 	function vary(n) {
 27 		return (Math.random() * n) - n / 2;
 28 	};
 29 	
 30 	// (x,y) to (r,theta)
 31 	function cartesianToPolar(x,y){
 32 		var theta = Math.atan(y/x);
 33 		var r = Math.sqrt(x*x + y*y);
 34 		return {
 35 			r: r, 
 36 			theta: theta
 37 		};
 38 	}
 39 	
 40 	// (r,theta) to (x,y)
 41 	function polarToCartesian(r,theta) {
 42 		var x = r * Math.cos(theta);
 43 		var y = r * Math.sin(theta);
 44 		return {
 45 			x: x,
 46 			y: y
 47 		};
 48 	}
 49 	
 50 	function varyVector(vector, angleVariance, magnitudeVariance) {
 51 		var polarCoords = cartesianToPolar(vector.x, vector.y);
 52 		polarCoords.r += vary(magnitudeVariance);
 53 		polarCoords.theta += vary(angleVariance);
 54 		var adjustedCoords = polarToCartesian(polarCoords.r, polarCoords.theta);
 55 		return new ex.base.Vector(adjustedCoords.x, adjustedCoords.y);
 56 	};
 57 	
 58 	function vectorDifference(a, b){
 59 		return new ex.base.Vector(b.x - a.x, b.y - a.y);
 60 	}
 61 	
 62 	function extend(object, extend) {
 63 		for(var param in extend) {
 64 			object[param] = extend[param];
 65 		}
 66 	};
 67 	
 68 	/*
 69 	 * Emitter Class
 70 	 */
 71 	ex.define("ex.plugins.Emitter2", {
 72 		constructor: function(emitterOptions, particleOptions) {
 73 			/* Set Emitter default options and extend new options */
 74 			var defaults = {
 75 				position: new ex.base.Vector(300, 300),			// position of emitter
 76 				particleVector: new ex.base.Vector(150, -50),	// base particle vector
 77 				angleVariance: Math.PI / 4,						// theta variance (radians)
 78 				magnitudeVariance: 0.2,							// v0 variance (0.0 - 1.0)
 79 				spawnSpeed: 5,									// max spawn per time step
 80 				maxParticles: 500,								// particle cap
 81 				sizeVariance: 5,								// randomness of size
 82 				lifeVariance: 20,								// randomness of lifespan
 83 				active: true									// is emitter emitting?
 84 			};
 85 			ex.extend(defaults, emitterOptions);
 86 			this.options = defaults;
 87 			this.options.origin = this.options.position.clone();		//remembers start for movement
 88 			this.options.lastPosition = this.options.origin.clone();	//used to detect movement
 89 			/* Set particle default options and extend new options */
 90 			var particleDefaults = {
 91 				position: this.options.position.clone(),	//.clone() to pass by value
 92 				vector: defaults.particleVector.clone(),	//
 93 				age: 0,
 94 				lifespan: 2,
 95 				size: 15,
 96 				alpha: 1,
 97 				color: '#ff0',
 98 				onDraw: function(particle) {
 99 					var y = -this.age * 100;
100 					y = Math.floor(y);
101 					particle.size *= 0.98;
102 					particle.color = this.color;
103 					particle.alpha = 0.5 - (particle.age / particle.lifespan * 0.4);
104 			    }
105 			};
106 			ex.extend(particleDefaults, particleOptions);
107 			this.particleOptions = particleDefaults;
108 			
109 			this.particles = [];
110 			this.active = true;
111 		},
112 		
113 		/**
114 		 * Two implementations for updating can be used, one favors accuracy, the other speed.
115 		 * 
116 		 * Accuracy (3 passes):
117 		 *  - remove dead particles
118 		 *  - spawn new particles
119 		 *  - update all particles
120 		 *  
121 		 * Speed (2 passes):
122 		 *  - update all particles, dead particles are removed
123 		 *  - spawn new particles (they miss their first time step)
124 		 *  
125 		 *  Below is the Speed implementation.
126 		 * @param dt
127 		 */
128 		update: function(dt) {
129 			// Update each live particle, remove dead ones
130 			var index = this.particles.length;
131 			var emitterMoved = false;
132 			if(this.options.position != this.options.lastPosition) {
133 				this.particleOptions.position = this.options.position.clone();
134 				this.particleOptions.vector = this.options.particleVector.clone();
135 				this.options.lastPosition = this.options.position.clone();
136 			}
137 			while(index--) {		
138 				if(this.particles[index].age > this.particles[index].lifespan) {
139 					this.particles.splice(index, 1);
140 				} else {
141 					this.particles[index].update(dt);
142 				}
143 			}
144 			
145 			// spawn new particles
146 			for(var birthCount = 0; birthCount < this.options.spawnSpeed; birthCount++) {
147 				if(this.particles.length >= this.options.maxParticles || this.options.active == false) {
148 					// Do nothing, no room or emitter inactive
149 					// break;
150 				} else {
151 					this.spawnParticle();
152 				}
153 			}
154 		},
155 		
156 		spawnParticle: function() {
157 			var optionsWithVariance = {};
158 			//MUST use .clone(), will pass by reference otherwise
159 			ex.extend(optionsWithVariance, ex.clone(this.particleOptions));
160 			//optionsWithVariance.lifespan += vary(this.options.lifeVariance);
161 			optionsWithVariance.vector = varyVector(optionsWithVariance.vector, this.options.angleVariance, this.options.magnitudeVariance);
162 			optionsWithVariance.size += vary(this.options.sizeVariance);
163 			this.particles.push(new ex.plugins.Particle2(optionsWithVariance));
164 		},
165 		
166 		render: function(context, camX, camY) {
167 			var index = this.particles.length;
168 			while(index--) {
169 				this.particles[index].render(context, camX, camY);
170 			}
171 		}
172 	});
173 });
