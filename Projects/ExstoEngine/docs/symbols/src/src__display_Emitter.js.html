1 ex.using([
  2   "ex.base.Vector",
  3   "ex.display.Particle"
  4 ], function () {
  5 	/*
  6 	 * Requirements:
  7 	 * - Support for animated sprites as particles
  8 	 * - drawing objects as particles
  9 	 * - variance in all directions (angle or cartesian)
 10 	 * - variance in age/size
 11 	 * - spawn rate
 12 	 * - initial velocity
 13 	 * - degree of randomness
 14 	 * - simulation and render phases
 15 	 * - emitters on points, lines, and maybe polys? (lines could be used for rain effects)
 16 	 */
 17 	
 18 	
 19 	ex.define("ex.display.Emitter", {
 20 		constructor: function(options) {
 21 			var defaults = {
 22 				position: new ex.base.Vector(150, 150),
 23 			    velocity: new ex.base.Vector(0, -3),
 24 			    xVariance: 20,
 25 			    yVariance: 5,
 26 			    spawnSpeed: 4,
 27 			    time: 100000,
 28 			    maxParticles: 500,
 29 			    size: 20,
 30 			    sizeVariance: 5,
 31 			    life: 30,
 32 			    lifeVariance: 10,    
 33 			    direction: 0,
 34 			    directionVariance: 15,
 35 			    color: '#cef',
 36 			    opacity: 1,
 37 			    scrollFactorX: 1,
 38 			    scrollFactorY: 1,
 39 			    onDraw: function(particle) {
 40 			      var y = -this.age * 3;
 41 			      particle.size *= 0.98;
 42 			      particle.color = 'rgb(255, ' + (y + 255) + ', 68)';
 43 			      particle.alpha = 0.5 - (particle.age / particle.life * 0.4);
 44 			    }
 45 		   };
 46 		   
 47 		   defaults.extend(options);
 48 		   
 49 		   this.options = defaults;
 50 		   
 51 		   this.particles = [];
 52 		   
 53 		   this.active = true;
 54 		},
 55 		
 56 		update: function(dt) {
 57 			// Subtract time and check for active
 58 			this.options.time -= dt;
 59 			if(this.options.time <= 0) {
 60 				this.active = false;
 61 			}
 62 			
 63 			// Update existing and remove dead particle
 64 			var i = this.particles.length;
 65 			while(i--) {
 66 				var particle = this.particles[i];
 67 				
 68 				if(particle.active == false) {
 69 					this.particles.splice(i, 1);
 70 				} else {
 71 					particle.update(dt);
 72 				}
 73 			}
 74 			
 75 			// Generate new particles
 76 			for(var spawned = 0; spawned < this.options.spawnSpeed; spawned++) {
 77 				if(this.particles.length >= this.options.maxParticles || this.options.time <= 0) {
 78 					// Do nothing
 79 				} else {
 80 					this.particles.push(new ex.display.Particle(this.options));
 81 				}
 82 			}
 83 		},
 84   
 85 		render: function(context, camX, camY) {
 86 			for(var i = 0; i < this.particles.length; i++) {
 87 				this.particles[i].render(context, camX, camY);
 88 			}
 89 		}
 90 	});	
 91 });
