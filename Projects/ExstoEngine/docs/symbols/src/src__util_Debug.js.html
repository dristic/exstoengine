  4 	function average(array) {
  5 		var ret = 0;
  6 		
  7 		var i = array.length;
  8 		while(i--) {
  9 			ret += array[i];
 10 		}
 11 		
 12 		ret = ret / array.length;
 13 		
 14 		return ret;
 15 	}
