1 /**
  2  * @class The collection of all items at a certain z-index in a map.
  3  * 
  4  * @param name
  5  *            {String}: the name of the layer (mostly for labeling in the IDE)
  6  * @param items
  7  *            {Array}: an array of all items in the layer
  8  * @param visible
  9  *            {Boolean}: allows rendering if true
 10  * @param opacity
 11  *            {Number}: ranges from 0.0 - 1.0, this is the transparency of the
 12  *            layer
 13  * @param position
 14  *            {ex.base.Point}: the starting location of the layer, used for
 15  *            rendering
 16  * @param scrollFactor
 17  *            {ex.base.Vector}: the multiplier for the amount of scroll per
 18  *            pixel moved by the camera, can be used to simulate depth of field.
 19  */
 20 ex.using([ 'ex.base.Point', 'ex.base.Vector' ], function() {
 21 	ex.define("ex.simplex.Layer", {
 22 
 23 		/**
 24 		 * creates a new layer with base attributes
 25 		 * 
 26 		 * @param $name
 27 		 *            {String}: name of the layer (mostly for IDE purposes)
 28 		 * @param $origin
 29 		 *            {ex.base.Point}: origin of the layer, defaults to (0,0) if
 30 		 *            not supplied
 31 		 * @param $scrollFactor
 32 		 *            {ex.base.Vector}: a multiplier that affects the scroll
 33 		 *            rate relative to the camera
 34 		 * @constructor
 35 		 */
 36 		constructor : function($name, $origin, $scrollFactor) {
 37 			this.name = $name;
 38 			this.items = [];
 39 			this.visible = true;
 40 			this.opacity = 1.0;
 41 
 42 			if ($origin == null) {
 43 				this.position = new ex.base.Point(0, 0);
 44 			} else {
 45 				this.position = $origin;
 46 			}
 47 
 48 			if ($scrollFactor == null) {
 49 				this.scrollFactor = new ex.base.Vector(1, 1);
 50 			} else {
 51 				this.scrollFactor = $scrollFactor;
 52 			}
 53 		},
 54 
 55 		/**
 56 		 * adds an item to the layer, could be anything really...
 57 		 * 
 58 		 * @param $item
 59 		 *            {Object}: something that belongs in the layer
 60 		 */
 61 		addItem : function($item) {
 62 			$item.visible = this.visible;
 63 			$item.opacity = this.opacity;
 64 			this.items.push($item);
 65 		},
 66 
 67 		/**
 68 		 * performs actions every time period dt
 69 		 * 
 70 		 * @param $dt
 71 		 *            {Number}: delta time, length of each time cycle
 72 		 */
 73 		update : function($dt) {
 74 
 75 		},
 76 
 77 		/**
 78 		 * checks properties and determines if layer is visible
 79 		 * 
 80 		 * @returns {Boolean}
 81 		 */
 82 		isVisible : function() {
 83 			if (this.visible && this.opacity > 0.0) {
 84 				return true;
 85 			} else {
 86 				return false;
 87 			}
 88 		},
 89 
 90 		/**
 91 		 * toggles visibility of the layer
 92 		 */
 93 		toggleVisibility: function() {
 94 			if(this.visible)
 95 				this.hide();
 96 			else
 97 				this.show();
 98 		},
 99 		
100 		/**
101 		 * sets visible property to true, does not affect opacity.
102 		 */
103 		show : function() {
104 			this.visible = true;
105 		},
106 
107 		/**
108 		 * sets visible property to false, does not affect opacity.
109 		 */
110 		hide : function() {
111 			this.visible = false;
112 		},
113 
114 		/**
115 		 * Supplies a canvas context and camera offset to each item and calls
116 		 * their render functions
117 		 * 
118 		 * @param $context
119 		 *            {Context}: canvas context to draw with
120 		 * @param $camX
121 		 *            {Number}: camera offset on x
122 		 * @param $camY
123 		 *            {Number}: camera offset on y
124 		 */
125 		render : function($context, $camX, $camY) {
126 			if (!this.isVisible()) // Don't render if it won't be seen
127 				return;
128 
129 			var count = this.items.length;
130 			while (count--) {
131 				this.items[count].render($context, $camX * this.scrollFactor.x, $camY * this.scrollFactor.y);
132 			}
133 		}
134 	});
135 });
