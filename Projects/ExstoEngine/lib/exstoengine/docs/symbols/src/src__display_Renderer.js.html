1 (function() {
  2 	ex.define("ex.display.Renderer", {
  3 		/**
  4 		 * The rendering component of the engine.
  5 		 * 
  6 		 * @name ex.display.Renderer
  7 		 * 
  8 		 * @param {Number} width
  9 		 * @param {Number} height
 10 		 * @param {String} bgColor hex value, eg #FFFFFF
 11 		 * @param {Canvas} canvas DOM Canvas to render to
 12 		 * 
 13 		 * @property {Canvas} canvas
 14 		 * @property {Context} context
 15 		 * @property {ex.display.Renderable[]} renderables objects to render
 16 		 */
 17 		constructor: function (width, height, bgColor, canvas) {
 18 			this.canvas = canvas || document.createElement("canvas");
 19 			this.canvas.id = "mainCanvas";
 20 			this.canvas.width = width;
 21 			this.canvas.height = height;
 22 			this.canvas.style.backgroundColor = bgColor;
 23 			this.context = this.canvas.getContext("2d");
 24 			
 25 			this.renderables = [];
 26 			
 27 			//--If a canvas was not passed in, add a new one to the page
 28 			if(canvas == null) {
 29 				document.body.appendChild(this.canvas);
 30 			}
 31 		},
 32 		
 33 		/**
 34 		 * Called on every frame to update the screen.
 35 		 * 
 36 		 * @function
 37 		 * @name update
 38 		 * @memberOf ex.display.Renderer
 39 		 * 
 40 		 * @param {Number} dt 
 41 		 * @param {Number} camX
 42 		 * @param {Number} camY
 43 		 */
 44 		update: function (dt, camX, camY) {			
 45 			this.context.clearRect(0, 0, this.canvas.width, this.canvas.height);
 46 			
 47 			var i = this.renderables.length;
 48 			while(i--) {
 49 				this.renderables[i].render(this.context, camX, camY);
 50 			}
 51 		},
 52 		
 53 		/**
 54 		 * Resizes the canvas.
 55 		 * 
 56 		 * @function
 57 		 * @name resizeCanvas
 58 		 * @memberOf ex.display.Renderer
 59 		 * 
 60 		 * @param {Number} newWidth
 61 		 * @param {Number} newHeight
 62 		 */
 63 		resizeCanvas: function (newWidth, newHeight) {
 64 			this.canvas.width = newWidth;
 65 			this.canvas.height = newHeight;
 66 		}
 67 	});
 68 }());
