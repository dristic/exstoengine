1 /**
  2  * @class A single environment that can be loaded into the world. In most cases,
  3  *        can be considered one game level or zone.
  4  * @param name
  5  *            {String}: name of map
  6  * @param layers
  7  *            {Array}: contains all layers in order of z-index
  8  */
  9 ex.using([ 'ex.simplex.Layer' ], function() {
 10 	ex.define("ex.simplex.Map", {
 11 		constructor : function($name) {
 12 			this.name = $name;
 13 			this.layers = [];
 14 		},
 15 
 16 		/**
 17 		 * Adds a layer to the map with the supplied items
 18 		 * 
 19 		 * @param $name
 20 		 *            {String}: name of the layer
 21 		 * @param $items
 22 		 *            {Array}: array of items to be included in the layer
 23 		 * @constructor
 24 		 */
 25 		addLayer : function($name, $items) {
 26 			if ($name != null && $items != null) {
 27 				var newLayer = new ex.simplex.Layer($name);
 28 				newLayer.items = $items;
 29 				this.layers.push(newLayer);
 30 			}
 31 		},
 32 
 33 		/**
 34 		 * toggles layer visiblity
 35 		 * 
 36 		 * @param $layerId
 37 		 *            {Number}: index of layer to toggle
 38 		 */
 39 		toggleLayer : function($layerId) {
 40 			var layer = this.layers[$layerId];
 41 			layer.visible = !layer.visible;
 42 		},
 43 
 44 		/**
 45 		 * performs actions every time period dt
 46 		 * 
 47 		 * @param $dt
 48 		 *            {Number}: delta time, length of each time cycle
 49 		 */
 50 		update : function($dt) {
 51 
 52 		},
 53 
 54 		/**
 55 		 * Supplies a canvas context and camera offset to each layer and calls
 56 		 * their render functions
 57 		 * 
 58 		 * @param $context
 59 		 *            {Context}: canvas context to draw with
 60 		 * @param $camX
 61 		 *            {Number}: camera offset on x
 62 		 * @param $camY
 63 		 *            {Number}: camera offset on y
 64 		 */
 65 		render : function($context, $camX, $camY) {
 66 			var i = this.layers.length;
 67 			while (i--) {
 68 				this.layers[i].render($context, $camX, $camY);
 69 			}
 70 		}
 71 	});
 72 });
