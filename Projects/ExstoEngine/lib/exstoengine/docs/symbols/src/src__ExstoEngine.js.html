343 	 * The base class definition for all classes in ExstoEngine. This function
344 	 * chains for proper use of instanceof and also supplies a _super method for
345 	 * accessing base classes.
346 	 * 
347 	 * @param namespace
348 	 *            The namespace of the new class
349 	 * @param base
350 	 *            The base class type to extend from or the class definition if
351 	 *            not extending

356 	ex.define = function(namespace, base, extension) {
357 		if (typeof namespace != 'string') {
358 			throw new Error(
359 					"Cannot create class, namespace must be of type String.\n"
360 							+ base.constructor);
361 			return null;
362 		}
363 		if (extension == null) {
364 			createNewClass(namespace, base);
365 		} else {
366 			extendBaseClass(namespace, base, extension);
367 		}
368 		
369 		this._defined.push(namespace);
370 		
371 		// Load all classes that require this class
372 		if(typeof this._namespaces[namespace] != 'undefined') {
373 			var functions = this._namespaces[namespace];
374 			var i = functions.length;
375 			while(i--) {
376 				var func = functions[i];
377 				ex.Array.remove(this._classes[func], namespace);
378 				if(this._classes[func].length == 0) {
379 					func();
380 					delete this._classes[func];
381 				}
382 				
383 				ex.Array.remove(this._namespaces[namespace], func);
384 				if(this._namespaces[namespace].length == 0) {
385 					delete this._namespaces[namespace];
386 				}
387 			}
388 		}

397 		function NewClass() {
398 			// empty function to hold new class
399 		};
400 		// --If a constructor is supplied call it
401 		if (typeof base.constructor == "function" && base.constructor != Object) {

408 	/**
409 	 * creates a new class with inheritance
410 	 * @param namespace {string} namespace of new class
411 	 * @param base {Object} base class
412 	 * @param extension {Object} base class extension (new class)
413 	 */
414 	function extendBaseClass(namespace, base, extension) {
415 		if (base == null) {
416 			throw new Error("The base class has not been defined: "
417 					+ extension.constructor);
418 		}
419 
420 		var _base = ex.clone(base.prototype);
421 		function NewClass() {
422 			// empty function to hold new class
423 			base.apply(this);
424 		};
425 
426 		// Set constructor: Priority on extension, then base class
427 		if (typeof extension.constructor == "function"
428 				&& extension.constructor != Object) {
429 			NewClass = extension.constructor;
430 		} else if (typeof _base.constructor == "function") {
431 			NewClass.constructor = _base.constructor;
432 		}
433 
434 		// clone, mix-in, set _super call, and add to namespace
435 		NewClass.prototype = _base;
436 		ex.extend(NewClass.prototype, extension);
437 		NewClass.prototype._super = function(func, args) {
438 			base.prototype[func].apply(this, args);
439 		};
440 		
441 		generateNamespace(namespace, NewClass);
442 	};

444 	/**
445 	 * Generates the namespace tree in the DOMWindow for the
446 	 * new class.
447 	 * @param namespace {string} namespace of class
448 	 * @param newClass {Object} class to be added to namespace
449 	 */
450 	function generateNamespace(namespace, newClass){
451 		var context = window;
452 		var parts = namespace.split('.');
453 		for ( var index = 0; index < parts.length; index++) {
454 			var part = parts[index];
455 			
456 			context[part] = context[part] || {};
457 			if (index == parts.length - 1) {
458 				context[part] = newClass;
459 				if('statics' in newClass.prototype) {
460 					ex.extend(context[part], newClass.prototype.statics);
461 				}
462 			}
463 			
464 			context = context[part];
465 		}
