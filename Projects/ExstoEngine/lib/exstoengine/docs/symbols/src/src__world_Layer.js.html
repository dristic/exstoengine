1 ex.using([ 
  2 	'ex.base.Point', 
  3 	'ex.base.Vector' 
  4 ], function() {
  5 	ex.define("ex.world.Layer", {
  6 
  7 		/**
  8 		 * A collection of items at a certain z-index in a map.
  9 		 * 
 10 		 * @name ex.world.Layer
 11 		 * 
 12 		 * @param {String} name name of the layer (mostly for IDE purposes)
 13 		 * @param {ex.display.SpriteMap} map tile data for the layer
 14 		 * @param {ex.base.Point} origin origin of the layer, defaults to (0,0) 
 15 		 * 		if not supplied
 16 		 * @param {ex.base.Vector} scrollFactor a multiplier that affects the 
 17 		 * 		scroll rate relative to the camera
 18 		 * 
 19 		 * @property {String} name the name of the layer (mostly for 
 20 		 * 		labeling in the IDE)
 21 		 * @property {Object[]} items an array of all items in the layer
 22 		 * @property {Boolean} visible allows rendering if true
 23 		 * @property {Number} opacity ranges from 0.0 - 1.0, this is the 
 24 		 * 		transparency of the layer
 25 		 * @property {ex.base.Vector} position the starting location of the 
 26 		 * 		layer
 27 		 * @property {ex.base.Vector} scrollFactor the multiplier for the 
 28 		 * 		amount of scroll per pixel moved by the camera, can be 
 29 		 * 		used to simulate depth of field.
 30 		 * 
 31 		 * @constructor
 32 		 */
 33 		constructor : function(name, map, origin, scrollFactor) {
 34 			this.name = name;
 35 			this.items = [];
 36 			this.map = map;
 37 			if(this.map != null) {
 38 				this.items.push(this.map);
 39 			}
 40 			this.visible = true;
 41 			this.opacity = 1.0;
 42 
 43 			if (origin == null) {
 44 				this.position = new ex.base.Point(0, 0);
 45 			} else {
 46 				this.position = origin;
 47 			}
 48 
 49 			if (scrollFactor == null) {
 50 				this.scrollFactor = new ex.base.Vector(1, 1);
 51 			} else {
 52 				this.scrollFactor = scrollFactor;
 53 			}
 54 		},
 55 
 56 		/**
 57 		 * Adds an item to the layer, could be anything really...
 58 		 * 
 59 		 * @function
 60 		 * @name addItem
 61 		 * @memberOf ex.world.Layer
 62 		 * 
 63 		 * @param {Object} item something that belongs in the layer
 64 		 */
 65 		addItem : function(item) {
 66 			if(item instanceof ex.world.CollisionMap){
 67 				if(this.map == null){
 68 					this._setMap(item);
 69 				} else {
 70 					console.error("Layers cannot contain more than one CollisionMap");
 71 				}
 72 			} else {
 73 				item.visible = this.visible;
 74 				item.opacity = this.opacity;
 75 				this.items.push(item);
 76 			}
 77 		},
 78 		
 79 		/**
 80 		 * Retrieves an item by name.
 81 		 * 
 82 		 * @function
 83 		 * @name getItem
 84 		 * @memberOf ex.world.Layer
 85 		 * 
 86 		 * @param {String} name name of the item to retrieve
 87 		 * 
 88 		 * @returns {Object} returns the item if found, otherwise null
 89 		 */
 90 		getItem: function(name) {
 91 			var index = this.items.length;
 92 			while(index--) {
 93 				if(this.items[index].name == name){
 94 					return this.items[index];
 95 				}
 96 			}
 97 		},
 98 		
 99 		_setMap : function(map) {
100 			this.map = map;
101 			this.items.push(this.map);
102 		},
103 		
104 		/**
105 		 * removes an item from the layer
106 		 * 
107 		 * @function
108 		 * @name removeItem
109 		 * @memberOf ex.world.Layer
110 		 * 
111 		 * @param {String} name name of item to remove
112 		 */
113 		removeItem: function(name) {
114 			var index = this.items.length;
115 			while(index--) {
116 				if(this.items[index].name == name) {
117 					this.items.splice(index, 1);
118 				}
119 			}
120 		},
121 
122 		/**
123 		 * Performs actions every time period dt
124 		 * 
125 		 * @function
126 		 * @name update
127 		 * @memberOf ex.world.Layer
128 		 * 
129 		 * @param {Number} dt delta time, length of each time cycle
130 		 */
131 		update : function(dt) {
132 			var index = this.items.length;
133 			while(index--){
134 				this.items[index].update(dt);
135 			}
136 		},
137 
138 		/**
139 		 * Checks properties and determines if layer is visible
140 		 * 
141 		 * @function
142 		 * @name isVisible
143 		 * @memberOf ex.world.Layer
144 		 * 
145 		 * @returns {Boolean}
146 		 */
147 		isVisible : function() {
148 			if (this.visible && this.opacity > 0.0) {
149 				return true;
150 			} else {
151 				return false;
152 			}
153 		},
154 
155 		/**
156 		 * Toggles visibility of the layer.
157 		 * 
158 		 * @function
159 		 * @name toggleVisibility
160 		 * @memberOf ex.world.Layer
161 		 */
162 		toggleVisibility : function() {
163 			if (this.visible)
164 				this.hide();
165 			else
166 				this.show();
167 		},
168 
169 		/**
170 		 * Sets visible property to true, does not affect opacity.
171 		 * 
172 		 * @function
173 		 * @name show
174 		 * @memberOf ex.world.Layer
175 		 */
176 		show : function() {
177 			this.visible = true;
178 		},
179 
180 		/**
181 		 * Sets visible property to false, does not affect opacity.
182 		 * 
183 		 * @function
184 		 * @name hide
185 		 * @memberOf ex.world.Layer
186 		 */
187 		hide : function() {
188 			this.visible = false;
189 		},
190 
191 		/**
192 		 * Supplies a canvas context and camera parameters for
193 		 * rendering to the canvas.
194 		 * 
195 		 * @function
196 		 * @name render
197 		 * @memberOf ex.world.Layer
198 		 * 
199 		 * @param {Context} context canvas context to draw with
200 		 * @param {Number} camX camera offset on x
201 		 * @param {Number} camY camera offset on y
202 		 * @param {Number} camWidth viewport width
203 		 * @param {Number} camHeight viewport height
204 		 */
205 		render : function(context, camX, camY, camWidth, camHeight) {
206 			if (!this.isVisible()) // Don't render if it won't be seen
207 				return;
208 			
209 			// render items
210 			var count = this.items.length;
211 			while (count--) {
212 				this.items[count].render(
213 						context, 
214 						camX * this.scrollFactor.x,
215 						camY * this.scrollFactor.y, 
216 						camWidth, 
217 						camHeight);
218 			}
219 		},
220 		
221 		debugRender: function (context, camX, camY, camWidth, camHeight) {
222 			this.render(context, camX, camY);
223 			var count = this.items.length;
224 			while (count--) {
225 				this.items[count].debugRender(
226 						context, 
227 						camX * this.scrollFactor.x, 
228 						camY * this.scrollFactor.y,
229 						camWidth, 
230 						camHeight);
231 			}
232 		}
233 	});
234 });
