1 ex.using([
  2       'ex.event.EventTarget'
  3 ], function () {
  4 	ex.define('ex.novus.NovusClient', 'ex.event.EventTarget', {
  5 		/**
  6 		 * The client object to connect to a Novus node.js server.
  7 		 * 
  8 		 * @name ex.novus.NovusClient
  9 		 * 
 10 		 * @param {String} url The url to connect to.
 11 		 * 
 12 		 * @property {String} url The url of the server connected to.
 13 		 * @property {Object} socket The socket.io socket connection.
 14 		 * 
 15 		 * @constructor
 16 		 */
 17 		constructor: function (url) {
 18 			this.url = url;
 19 			this.socket = io.connect(url);
 20 			this.listeners = {};
 21 			
 22 			var that = this;
 23 			
 24 			this.socket.on('login', function (data) {
 25 				that.callback(data.success);
 26 			});
 27 			
 28 			this.socket.on('roomList', function (data) {
 29 				that.callback(data.list);
 30 			});
 31 			
 32 			this.socket.on('joinRoom', function (data) {
 33 				that.callback(data);
 34 			});
 35 			
 36 			this.socket.on('createRoom', ex.bind(this, function (data) {
 37 				this.dispatchEvent('createRoom', data);
 38 			}));
 39 			
 40 			this.socket.on('roomMessage', ex.bind(this, function(data) {
 41 				var callbacks = this.listeners[data.type] || [];
 42 				callbacks.forEach(function (callback, index, array) {
 43 					callback(data.message);
 44 				});
 45 			}));
 46 		},
 47 		
 48 		/**
 49 		 * Allows a client to listen for a certain room message type and call a function
 50 		 * when it receives that message.
 51 		 * 
 52 		 * @param {String} type The type to listen for.
 53 		 * @param {Function} callback The function to call when we receive that message.
 54 		 */
 55 		on: function(message, callback) {
 56 			if(typeof this.listeners[message] == "undefined") {
 57 				this.listeners[message] = [];
 58 			}
 59 			
 60 			this.listeners[message].push(callback);
 61 		},
 62 		
 63 		/**
 64 		 * Logs a user into the novus server using the given credentials.
 65 		 * 
 66 		 * Callback: void ({Boolean} success)
 67 		 * 
 68 		 * @param {String} name The user name to login as.
 69 		 * @param {String} password The user's password.
 70 		 * @param {Function} callback What to call upon logging in.
 71 		 */
 72 		login: function (name, password, callback) {
 73 			this.callback = callback;
 74 			this.socket.emit('login', { name: name, password: password });
 75 		},
 76 		
 77 		roomList: function (callback) {
 78 			this.callback = callback;
 79 			this.socket.emit('roomList', {});
 80 		},
 81 		
 82 		/**
 83 		 * Creates and joins a room on the server.
 84 		 * 
 85 		 * @param {String} name The name of the room. Must be unique.
 86 		 */
 87 		createRoom: function(name) {
 88 			this.socket.emit('createRoom', { name: name });
 89 		},
 90 		
 91 		/**
 92 		 * Joins a room on the server.
 93 		 * 
 94 		 * @param {String} The name of the room to join.
 95 		 */
 96 		joinRoom: function(name, callback) {
 97 			this.callback = callback;
 98 			this.socket.emit('joinRoom', { name: name });
 99 		},
100 		
101 		/**
102 		 * Leaves the current room if the client is currently in one.
103 		 */
104 		leaveRoom: function() {
105 			this.socket.emit('leaveRoom', {});
106 		},
107 		
108 		/**
109 		 * Sends a message to every user in the same room.
110 		 * 
111 		 * @param {String} type The type of the message to send. This is the type that NovusClient.on looks for.
112 		 * @param {Object} message The data message to send.
113 		 */
114 		message: function(type, message) {
115 			this.socket.emit('roomMessage', { type: type, message: message });
116 		},
117 		
118 		/**
119 		 * Sends a message to the given username.
120 		 * 
121 		 * @param {String} name The user to send the message to.
122 		 * @param {Type} type The type of the message to send.
123 		 * @param {Object} message The data message to send.
124 		 */
125 		messageTo: function (name, type, message) {
126 			this.socket.emit('roomMessageTo', { name: name, type: type, message: message });
127 		}
128 	});
129 });
