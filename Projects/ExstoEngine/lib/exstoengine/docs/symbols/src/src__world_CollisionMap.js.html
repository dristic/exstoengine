1 ex.using([
  2   "ex.base.Point",
  3   "ex.world.TileMap"
  4 ], function () {
  5 	ex.define("ex.world.CollisionMap", ex.world.TileMap, {
  6 		/**
  7 		 * Tile based collision data for an area.
  8 		 * 
  9 		 * @name ex.world.CollisionMap
 10 		 * 
 11 		 * @param tileWidth
 12 		 * @param tileHeight
 13 		 * @param data
 14 		 * @param debug
 15 		 * @param debugImages
 16 		 * 
 17 		 * @property tileWidth
 18 		 * @property tileHeight
 19 		 * @property collides
 20 		 * @property position
 21 		 * @property scrollFactor
 22 		 * 
 23 		 * @constructor
 24 		 */
 25 		constructor: function(tileWidth, tileHeight, data, debug, debugImages) {
 26 			this.tileWidth = tileWidth;
 27 			this.tileHeight = tileHeight;
 28 			this.collides = true;
 29 			
 30 			this.type = "TileMap";
 31 			
 32 			this.position = new ex.base.Vector(0,0);
 33 			this.scrollFactor = new ex.base.Vector(1,1);
 34 			
 35 			this._super("constructor", [tileWidth, tileHeight, data]);
 36 		},
 37 		
 38 		/**
 39 		 * 
 40 		 */
 41 		onCollide: function(){
 42 			
 43 		},
 44 		
 45 		/**
 46 		 * 
 47 		 * @param x
 48 		 * @param y
 49 		 * @returns
 50 		 */
 51 		getTile: function(x, y) {
 52 			x = Math.floor(x / this.tileWidth);
 53 			y = Math.floor(y / this.tileHeight);
 54 			
 55 			if(x >= 0 && y >= 0 &&
 56 					y < this.data.length && x < this.data[y].length) {
 57 				return this.data[y][x];
 58 			} else {
 59 				return null;
 60 			}
 61 		},
 62 		
 63 		/**
 64 		 * 
 65 		 * @param x
 66 		 * @param y
 67 		 * @param value
 68 		 */
 69 		setTile: function(x, y, value) {
 70 			var tile = this.getTile(x,y);
 71 			if(tile != null){
 72 				tile.value = value;
 73 			}
 74 		},
 75 		
 76 		update: function(dt) {
 77 			
 78 		},
 79 		
 80 		render: function(context, camX, camY){
 81 			// This has no render view
 82 		},
 83 		
 84 		/**
 85 		 * 
 86 		 */
 87 		debugRender: function(context, camX, camY) {
 88 			var tile = null;
 89 			var yPos = 0,
 90 				xPos = 0;
 91 			for(yPos = 0; yPos < this.data.length; yPos++){
 92 				for(xPos = 0; xPos < this.data[yPos].length; xPos++){
 93 					tile = this.data[yPos][xPos];
 94 					if(tile.value != 0){
 95 						this._renderTileEdges(context, camX, camY, tile);
 96 					}
 97 				}
 98 			}
 99 		},
100 	
101 		/**
102 		 * 
103 		 * @param context
104 		 * @param camX
105 		 * @param camY
106 		 * @param tile
107 		 */
108 		_renderTileEdges: function(context, camX, camY, tile) {
109 			if(tile.edges.left){
110 				context.drawImage(this.debugImages.left,
111 						tile.position.x - (camX * this.scrollFactor.x),
112 						tile.position.y - (camY * this.scrollFactor.y),
113 						tile.width,
114 						tile.height);
115 			}
116 			if(tile.edges.right){
117 				context.drawImage(this.debugImages.right,
118 						tile.position.x - (camX * this.scrollFactor.x),
119 						tile.position.y - (camY * this.scrollFactor.y),
120 						tile.width,
121 						tile.height);
122 			}
123 			if(tile.edges.top){
124 				context.drawImage(this.debugImages.top,
125 						tile.position.x - (camX * this.scrollFactor.x),
126 						tile.position.y - (camY * this.scrollFactor.y),
127 						tile.width,
128 						tile.height);
129 			}
130 			if(tile.edges.bottom){
131 				context.drawImage(this.debugImages.bottom,
132 						tile.position.x - (camX * this.scrollFactor.x),
133 				      	tile.position.y - (camY * this.scrollFactor.y),
134 				      	tile.width,
135 				      	tile.height);
136 			}
137 		}
138 	});
139 });
