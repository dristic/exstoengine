1 ex.using([
  2     "ex.world.Tile",
  3     "ex.base.Vector"
  4 ], function () {
  5 	ex.define("ex.world.TileMap", {
  6 		
  7 		/**
  8 		 * A 2D grid of ex.world.Tile objects used to create a game level.
  9 		 * 
 10 		 * @name ex.world.TileMap
 11 		 * 
 12 		 * @param {Number} tileWidth the uniform width of each tile.
 13 		 * 		Do not change once set, can and most likely will cause
 14 		 * 		odd behavior.
 15 		 * @param {Number} tileHeight the uniform height of each tile.
 16 		 * 		Do not change once set, can and most likely will cause
 17 		 * 		odd behavior.
 18 		 * @param {Number[][]} map the 2D numerical values for each tile.
 19 		 * 		These values will be converted into ex.world.Tile objects
 20 		 * 		during initialization.
 21 		 * 
 22 		 * @property {Number} tileWidth
 23 		 * @property {Number} tileHeight
 24 		 * @property {ex.world.Tile[][]} data 2D grid of ex.world.Tile objects
 25 		 * @property {Number} width total width of the map
 26 		 * @property {Number} height total height of the map
 27 		 * 
 28 		 * @constructor 
 29 		 */
 30 		constructor: function(tileWidth, tileHeight, map) {
 31 			this.tileWidth = tileWidth;
 32 			this.tileHeight = tileHeight;
 33 			this.data = convertValueMapToTileMap(map, tileWidth, tileHeight);
 34 			setupNeighbors(this.data);
 35 			
 36 			//--Calculate width and height
 37 			this.width = map[0].length * this.tileWidth;
 38 			this.height = map.length * this.tileHeight;
 39 		},
 40 		
 41 		/**
 42 		 * Retrieves a tile at the supplied x, y position.
 43 		 * 
 44 		 * @function
 45 		 * @name getTile
 46 		 * @memberOf ex.world.TileMap
 47 		 * 
 48 		 * @param {Number} x
 49 		 * @param {Number} y
 50 		 * @returns {ex.world.Tile}
 51 		 */
 52 		getTile: function(x, y) {
 53 			x = Math.floor(x / this.tileWidth);
 54 			y = Math.floor(y / this.tileHeight);
 55 			
 56 			if(x >= 0 && x < this.data[0].length && 
 57 					y >= 0 && y < this.data.length){
 58 				return this.data[y][x];
 59 			} else {
 60 				return null;
 61 			}
 62 		},
 63 		
 64 		/**
 65 		 * Sets a tile at the given x,y position to the given value.
 66 		 * 
 67 		 * @function
 68 		 * @name setTile
 69 		 * @memberOf ex.world.TileMap
 70 		 * 
 71 		 * @param {Number} x
 72 		 * @param {Number} y
 73 		 * @param {Number} value
 74 		 */
 75 		setTile: function(x, y, value) {
 76 			x = Math.floor(x / this.tileWidth);
 77 			y = Math.floor(y / this.tileHeight);
 78 			this.data[y][x].setValue(value);
 79 		}
 80 	});
 81 	
 82 	function convertValueMapToTileMap(tileData, tileWidth, tileHeight) {
 83 		var yPos = 0;
 84 		var xPos = 0;
 85 		for(yPos; yPos < tileData.length; yPos++) {
 86 			for(xPos = 0; xPos < tileData[yPos].length; xPos++) {
 87 				tileData[yPos][xPos] = new ex.world.Tile(
 88 						tileData[yPos][xPos],
 89 						new ex.base.Vector(xPos * tileWidth, yPos * tileHeight),
 90 						tileWidth,
 91 						tileHeight);
 92 			}
 93 		}
 94 		return tileData;
 95 	};
 96 	
 97 	function setupNeighbors(tiles) {
 98 		var yPos = 0;
 99 		var xPos = 0;
100 		for(yPos; yPos < tiles.length; yPos++) {
101 			for(xPos = 0; xPos < tiles[yPos].length; xPos++) {
102 				if(xPos == 0) {
103 					tiles[yPos][xPos].neighbors.left = null;
104 					tiles[yPos][xPos].neighbors.right = tiles[yPos][xPos + 1];
105 				} else if(xPos == tiles[yPos].length - 1) {
106 					tiles[yPos][xPos].neighbors.left = tiles[yPos][xPos - 1];
107 					tiles[yPos][xPos].neighbors.right = null;
108 				} else {
109 					tiles[yPos][xPos].neighbors.left = tiles[yPos][xPos - 1];
110 					tiles[yPos][xPos].neighbors.right = tiles[yPos][xPos + 1];
111 				}
112 				if(yPos == 0) {
113 					tiles[yPos][xPos].neighbors.top = null;
114 					tiles[yPos][xPos].neighbors.bottom = tiles[yPos + 1][xPos];
115 				} else if(yPos == tiles.length - 1) {
116 					tiles[yPos][xPos].neighbors.top = tiles[yPos - 1][xPos];
117 					tiles[yPos][xPos].neighbors.bottom = null;
118 				} else {
119 					tiles[yPos][xPos].neighbors.top = tiles[yPos - 1][xPos];
120 					tiles[yPos][xPos].neighbors.bottom = tiles[yPos + 1][xPos];
121 				}
122 				tiles[yPos][xPos].update();
123 			}
124 		}
125 	};
126 });
