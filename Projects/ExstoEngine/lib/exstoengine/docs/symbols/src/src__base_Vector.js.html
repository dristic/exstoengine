1 (function () {
  2 	ex.define("ex.base.Vector", {

  4 		 * @name ex.base.Vector
  5 		 * @param {Number} x magnitude on x axis
  6 		 * @param {Number} y magnitude on y axis

  8 		 * @property {Number} x
  9 		 * @property {Number} y

 14 			this.y = y || 0;

 27 			this.y += v.y;
 28 			return this;
 29 		},

 32 		 * Adds a scaled vector to this vector. To subtract,

 36 		 * @memberOf ex.base.Vector
 37 		 * @param {ex.base.Vector} v vector to add

 39 		 * @returns {ex.base.Vector} this
 40 		 */

 43 			this.y += (v.y * s);
 44 			return this;
 45 		},

 47 		/**

 53 		 * @returns {ex.base.Vector} this
 54 		 */

 67 		 * @returns {ex.base.Vector} this
 68 		 */
 69 		scale: function(s) {
 70 			this.x *= s;
 71 			this.y *= s;
 72 			return this;

105 		 * @memberOf ex.base.Vector
106 		 * @returns {ex.base.Vector} a copy of this
107 		 */

117 		 * @param {Number} rad radians to rotate by
118 		 * @returns {ex.base.Vector} this
119 		 */
120 		rotate: function(rad) {
