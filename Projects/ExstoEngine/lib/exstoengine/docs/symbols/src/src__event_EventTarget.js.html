  2 	
  3 	// Static functions for listening and unlistening to events
  4 	ex.event = {
  5 		/*
  6 		 * Listens to an event on a javascript object
  7 		 * or a DOM object
  8 		 */
  9 		listen: function (target, event, func, handler) {
 10 			// Bind to a different handler if needed
 11 			if(handler != null) {
 12 				func = ex.bind(handler, func);
 13 			}
 14 			
 15 			// Figure out what event handling the object supports
 16 			// and use it to listen on
 17 			if(target.attachEvent) {
 18 				target.attachEvent('on' + event, func);
 19 			} else if(target.addEventListener) {
 20 				target.addEventListener(event, func);
 21 			}
 22 		},
 23 		
 24 		/**
 25 		 * Automatically unlistens to an event after it has been fired
 26 		 * 
 27 		 * @param target {Object} The object to listen on.
 28 		 * @param event {String} The event to listen to.
 29 		 * @param func {Function} The function to call when the event happens.
 30 		 * @param handler {Object} [Optional] The object to bind to when calling the function.
 31 		 */
 32 		listenOnce: function (target, event, func, handler) {
 33 			var newFunc = function () {
 34 				ex.event.unlisten(target, event, func);
 35 				
 36 				func.call(this, arguments);
 37 			};
 38 			
 39 			ex.event.listen(target, event, newFunc, handler);
 40 		},
 41 		
 42 		unlisten: function (target, event, func) {
 43 			if(target.removeEventListener) {
 44 				target.removeEventListener(event, func);
 45 			}
 46 		}
 47 	};
 48 	
 49 	// The extendable event target object
 50     ex.define("ex.event.EventTarget", {
 51         constructor: function () {
 52         	this._listeners = {};
 53         },
 54         
 55         addEventListener: function (event, func) {
 56         	if(typeof this._listeners[event] == "undefined") {
 57         		this._listeners[event] = [];
 58         	}
 59         	
 60         	this._listeners[event].push(func);
 61         },
 62         
 63         /**
 64          * Dispatches all events with the selector event.
 65          * Returns true if events are dispatched, false
 66          * if there are no events for that selector.
 67          * 
 68          * @param event
 69          * @param params
 70          * @returns {Boolean}
 71          */
 72         dispatchEvent: function (event, eventObject) {
 73         	if(this._listeners[event] == null){
 74         		return false;
 75         	}
 76         	
 77         	var i = this._listeners[event].length;
 78         	while(i--) {
 79         		var func = this._listeners[event][i],
 80         			ret = func(eventObject);
 81         		if(ret == false) {
 82         			break;
 83         		}
 84         	}
 85         	return true;
 86         },
 87         
 88         removeEventListener: function (event, func) {
 89         	this._listeners[event].splice(this._listeners[event].indexOf(func), 1);
 90         }
 91     });
