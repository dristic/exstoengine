1 /**
  2  * @class The collection of all items at a certain z-index in a map.
  3  * 
  4  * 
  5  */
  6 ex.using([ 'ex.base.Point', 'ex.base.Vector' ], function() {
  7 	ex.define("ex.simplex.Layer", {
  8 
  9 		/**
 10 		 * A collection of items at a certain z-index in a map.
 11 		 * 
 12 		 * @name ex.simplex.Layer
 13 		 * 
 14 		 * @param {String} name name of the layer (mostly for IDE purposes)
 15 		 * @param {ex.display.SpriteMap} map tile data for the layer
 16 		 * @param {ex.base.Point} origin origin of the layer, defaults to (0,0) 
 17 		 * 		if not supplied
 18 		 * @param {ex.base.Vector} scrollFactor a multiplier that affects the 
 19 		 * 		scroll rate relative to the camera
 20 		 * 
 21 		 * @property {String} name the name of the layer (mostly for 
 22 		 * 		labeling in the IDE)
 23 		 * @property {Object[]} items an array of all items in the layer
 24 		 * @property {Boolean} visible allows rendering if true
 25 		 * @property {Number} opacity ranges from 0.0 - 1.0, this is the 
 26 		 * 		transparency of the layer
 27 		 * @property {ex.base.Vector} position the starting location of the 
 28 		 * 		layer
 29 		 * @property {ex.base.Vector} scrollFactor the multiplier for the 
 30 		 * 		amount of scroll per pixel moved by the camera, can be 
 31 		 * 		used to simulate depth of field.
 32 		 * 
 33 		 * @constructor
 34 		 */
 35 		constructor : function(name, map, origin, scrollFactor) {
 36 			this.name = name;
 37 			this.items = [];
 38 			this.map = map;
 39 			if(this.map != null) {
 40 				this.items.push(this.map);
 41 			}
 42 			this.visible = true;
 43 			this.opacity = 1.0;
 44 
 45 			if (origin == null) {
 46 				this.position = new ex.base.Point(0, 0);
 47 			} else {
 48 				this.position = origin;
 49 			}
 50 
 51 			if (scrollFactor == null) {
 52 				this.scrollFactor = new ex.base.Vector(1, 1);
 53 			} else {
 54 				this.scrollFactor = scrollFactor;
 55 			}
 56 		},
 57 
 58 		/**
 59 		 * Adds an item to the layer, could be anything really...
 60 		 * 
 61 		 * @function
 62 		 * @name addItem
 63 		 * @memberOf ex.simplex.Layer
 64 		 * 
 65 		 * @param {Object} item something that belongs in the layer
 66 		 */
 67 		addItem : function(item) {
 68 			if(item.type == "SpriteMap"){
 69 				if(this.map == null){
 70 					this._setMap(item);
 71 				} else {
 72 					console.error("Layers cannot contain more than one SpriteMap");
 73 				}
 74 			} else {
 75 				item.visible = this.visible;
 76 				item.opacity = this.opacity;
 77 				this.items.push(item);
 78 			}
 79 		},
 80 		
 81 		_setMap : function(map) {
 82 			this.map = map;
 83 			this.items.push(this.map);
 84 		},
 85 		
 86 		/**
 87 		 * removes an item from the layer
 88 		 * 
 89 		 * @function
 90 		 * @name removeItem
 91 		 * @memberOf ex.simplex.Layer
 92 		 * 
 93 		 * @param {Number} index index of the item to remove
 94 		 */
 95 		removeItem: function(index) {
 96 			if (index < 0 || index >= this.items.length) {
 97 				return;
 98 			}
 99 			this.items.splice($index, 1);
100 		},
101 
102 		/**
103 		 * Performs actions every time period dt
104 		 * 
105 		 * @function
106 		 * @name update
107 		 * @memberOf ex.simplex.Layer
108 		 * 
109 		 * @param {Number} dt delta time, length of each time cycle
110 		 */
111 		update : function(dt) {
112 			var index = this.items.length;
113 			while(index--){
114 				this.items[index].update(dt);
115 			}
116 		},
117 
118 		/**
119 		 * Checks properties and determines if layer is visible
120 		 * 
121 		 * @function
122 		 * @name isVisible
123 		 * @memberOf ex.simplex.Layer
124 		 * 
125 		 * @returns {Boolean}
126 		 */
127 		isVisible : function() {
128 			if (this.visible && this.opacity > 0.0) {
129 				return true;
130 			} else {
131 				return false;
132 			}
133 		},
134 
135 		/**
136 		 * Toggles visibility of the layer.
137 		 * 
138 		 * @function
139 		 * @name toggleVisibility
140 		 * @memberOf ex.simplex.Layer
141 		 */
142 		toggleVisibility : function() {
143 			if (this.visible)
144 				this.hide();
145 			else
146 				this.show();
147 		},
148 
149 		/**
150 		 * Sets visible property to true, does not affect opacity.
151 		 * 
152 		 * @function
153 		 * @name show
154 		 * @memberOf ex.simplex.Layer
155 		 */
156 		show : function() {
157 			this.visible = true;
158 		},
159 
160 		/**
161 		 * Sets visible property to false, does not affect opacity.
162 		 * 
163 		 * @function
164 		 * @name hide
165 		 * @memberOf ex.simplex.Layer
166 		 */
167 		hide : function() {
168 			this.visible = false;
169 		},
170 
171 		/**
172 		 * Supplies a canvas context and camera offset to each item and calls
173 		 * their render functions
174 		 * 
175 		 * @function
176 		 * @name render
177 		 * @memberOf ex.simplex.Layer
178 		 * 
179 		 * @param {Context} context canvas context to draw with
180 		 * @param {Number} camX camera offset on x
181 		 * @param {Number} camY camera offset on y
182 		 */
183 		render : function(context, camX, camY) {
184 			if (!this.isVisible()) // Don't render if it won't be seen
185 				return;
186 			
187 			// render items
188 			var count = this.items.length;
189 			while (count--) {
190 				this.items[count].render(context, camX * this.scrollFactor.x,
191 						camY * this.scrollFactor.y);
192 			}
193 		}
194 	});
195 });
