1 ex.using([ 
  2     'ex.base.Point', 
  3     'ex.base.Vector' 
  4 ], function() {
  5 	ex.define("ex.display.Image", {
  6 		/**
  7 		 * Container class for HTML Images with additional position,
  8 		 * size, and name data.
  9 		 * 
 10 		 * @name ex.display.Image
 11 		 * 
 12 		 * @param {Image} image
 13 		 * @param {ex.base.Point} position
 14 		 * @param {ex.base.Vector} size width and height
 15 		 * @param {String} name 
 16 		 * 
 17 		 * @property {Boolean} visible controls whether the image
 18 		 * 		is rendered or not.
 19 		 * @property {String} type For internal use only. Do not change!
 20 		 * @property {ex.base.Point} position
 21 		 * @property {ex.base.Vector} size width and height
 22 		 * @property {Image} image
 23 		 * 
 24 		 * @constructor
 25 		 */
 26 		constructor : function(image, position, size, name) {
 27 			this.visible = true;
 28 			this.type = "Image";
 29 			this.name = name;
 30 			this.position = position || new ex.base.Point(0, 0);
 31 			this.width = image.naturalWidth;
 32 			this.height = image.naturalHeight;
 33 			this.image = image;
 34 			
 35 			// If the image is not loaded reset size when it loads
 36 			if(this.image.complete == false) {
 37 				ex.event.listenOnce(this.image, 'load', this.autoSize, this);
 38 			}
 39 		},
 40 		
 41 		/**
 42 		 * Auto sizes this object's width and height to the image supplied, and removes
 43 		 * any event listeners on the image for loading
 44 		 * 
 45 		 * @function
 46 		 * @name autoSize
 47 		 * @memberOf ex.display.Image
 48 		 */
 49 		autoSize: function () {
 50 			this.width = this.image.naturalWidth;
 51 			this.height = this.image.naturalHeight;
 52 		},
 53 
 54 		/**
 55 		 * Retrieves an axis aligned bounding box (AABB) surrounding
 56 		 * 		the image.
 57 		 * 
 58 		 * @function
 59 		 * @name getBounds
 60 		 * @nemberOf ex.display.Image
 61 		 * 
 62 		 * @returns {ex.base.Rectangle} 
 63 		 */
 64 		getBounds: function() {
 65 			return new ex.base.Rectangle(
 66 					this.position,
 67 					this.size.x,
 68 					this.size.y);
 69 		},
 70 		
 71 		/**
 72 		 * performs actions every time period dt
 73 		 * 
 74 		 * @function
 75 		 * @name update
 76 		 * @memberOf ex.display.Image
 77 		 * 
 78 		 * @param {Number} dt length of current time cycle
 79 		 */
 80 		update : function(dt) {
 81 
 82 		},
 83 
 84 		/**
 85 		 * Supplies a canvas context and camera offset to each item and calls
 86 		 * their render functions
 87 		 * 
 88 		 * @function
 89 		 * @name render
 90 		 * @memberOf ex.display.Image
 91 		 * 
 92 		 * @param {Context} context canvas context to draw with
 93 		 * @param {Number} camX camera offset on x
 94 		 * @param {Number} camY camera offset on y
 95 		 * @param {Number} camWidth viewport width
 96 		 * @param {Number} camHeight viewport height
 97 		 */
 98 		render : function(context, camX, camY, camWidth, camHeight) {
 99 			if(!this.visible){
100 				return;
101 			}
102 			
103 			// Position of the sprite in the viewport
104 			var viewPortX = ex.toInt(this.position.x - (camX * this.scrollFactor.x)),
105 				viewPortY = ex.toInt(this.position.y - (camY * this.scrollFactor.y));
106 						
107 			// Do nothing if sprite is out of the viewport
108 			if((viewPortX + this.width) < 0
109 					&& viewPortX > camWidth
110 					&& (viewPortY + this.height) < 0
111 					&& viewPortY > camHeight) {
112 				return;
113 			}
114 			
115 			if (this.image == null) {
116 				context.fillStyle = '#888888';
117 				context.fillRect(
118 						this.position.x, this.position.y,
119 						this.size.x, this.size.y
120 				);
121 			} else {
122 				context.drawImage(
123 						this.image, 
124 						viewPortX, viewPortY, 
125 						this.size.x, this.size.y
126 				);
127 			}
128 		}
129 	});
130 });
