1 ex.using([
  2     "ex.base.Vector"          
  3 ], function () {
  4 	ex.define("ex.display.Camera", {
  5 		/**
  6 		 * A viewport for the game.
  7 		 * @name ex.display.Camera
  8 		 * 
  9 		 * @property {ex.base.Point} position Position of camera.
 10 		 * 		Initialized to (0,0).
 11 		 * @property {Number} width width of the viewport
 12 		 * @property {Number} height height of the viewport
 13 		 * @property {Object} following Entity being followed. 
 14 		 * 		Initialized to null.
 15 		 * @property {Canvas} canvas Canvas to use for rendering.
 16 		 * 		Initialized to null.
 17 		 * @constructor
 18 		 */
 19 		constructor: function(position, width, height) {
 20 			this.position = position;
 21 			this.width = width;
 22 			this.height = height;
 23 			this.following = null;
 24 			this.canvas = null;
 25 		},
 26 		
 27 		/**
 28 		 * Moves the camera to the position of the object being followed.
 29 		 * @function
 30 		 * @name update
 31 		 * @memberOf ex.display.Camera
 32 		 * @param {Number} dt
 33 		 */
 34 		update: function(dt) {
 35 			if(this.following != null && this.canvas != null) {
 36 				this.position.x = this.following.position.x + (this.following.width >> 1) - (this.canvas.width >> 1);
 37 				this.position.y = this.following.position.y + (this.following.height >> 1) - (this.canvas.height >> 1);
 38 			}
 39 		},
 40 		
 41 		/**
 42 		 * Translates the camera by the supplied x and y values.
 43 		 * @function
 44 		 * @name translate
 45 		 * @memberOf ex.display.AnimatedSprite
 46 		 * @param {Number} x translation on the x axis
 47 		 * @param {Number} y translation on the y axis
 48 		 */
 49 		translate: function(x, y) {
 50 			this.position.x += x;
 51 			this.position.y += y;
 52 		},
 53 		
 54 		/**
 55 		 * Moves the camera to a specific position.
 56 		 * @function
 57 		 * @name moveTo
 58 		 * @memberOf ex.display.AnimatedSprite
 59 		 * @param {Number} x position on the x axis
 60 		 * @param {Number} y position on the y axis
 61 		 */
 62 		moveTo: function (x, y) {
 63 			this.position.x = x;
 64 			this.position.y = y;
 65 		},
 66 		
 67 		/**
 68 		 * Sets an object for the camera to follow.
 69 		 * @function
 70 		 * @name follow
 71 		 * @memberOf ex.display.AnimatedSprite
 72 		 * @param {Entity} object entity to follow
 73 		 */
 74 		follow: function (object) {
 75 			this.following = object;
 76 		}
 77 	});
 78 });
