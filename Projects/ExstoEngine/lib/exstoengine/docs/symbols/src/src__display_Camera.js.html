1 ex.using([
  2     "ex.base.Vector"          
  3 ], function () {
  4 	ex.define("ex.display.Camera", {
  5 		/**
  6 		 * A viewport for the game.
  7 		 * @name ex.display.Camera
  8 		 * 
  9 		 * @property {ex.base.Point} position Position of camera.
 10 		 * 		Initialized to (0,0).
 11 		 * @property {Object} following Entity to follow. 
 12 		 * 		Initialized to null.
 13 		 * @property {Canvas} canvas Canvas to use for rendering.
 14 		 * 		Initialized to null.
 15 		 * @constructor
 16 		 */
 17 		constructor: function() {
 18 			this.position = new ex.base.Vector(0,0);
 19 			this.following = null;
 20 			this.canvas = null;
 21 		},
 22 		
 23 		/**
 24 		 * Moves the camera to the position of the object being followed.
 25 		 * @function
 26 		 * @name update
 27 		 * @memberOf ex.display.Camera
 28 		 * @param {Number} dt
 29 		 */
 30 		update: function(dt) {
 31 			if(this.following != null && this.canvas != null) {
 32 				this.position.x = this.following.position.x + (this.following.width >> 1) - (this.canvas.width >> 1);
 33 				this.position.y = this.following.position.y + (this.following.height >> 1) - (this.canvas.height >> 1);
 34 			}
 35 		},
 36 		
 37 		/**
 38 		 * Translates the camera by the supplied x and y values.
 39 		 * @function
 40 		 * @name translate
 41 		 * @memberOf ex.display.AnimatedSprite
 42 		 * @param {Number} x translation on the x axis
 43 		 * @param {Number} y translation on the y axis
 44 		 */
 45 		translate: function(x, y) {
 46 			this.position.x += x;
 47 			this.position.y += y;
 48 		},
 49 		
 50 		/**
 51 		 * Moves the camera to a specific position.
 52 		 * @function
 53 		 * @name moveTo
 54 		 * @memberOf ex.display.AnimatedSprite
 55 		 * @param {Number} x position on the x axis
 56 		 * @param {Number} y position on the y axis
 57 		 */
 58 		moveTo: function (x, y) {
 59 			this.position.x = x;
 60 			this.position.y = y;
 61 		},
 62 		
 63 		/**
 64 		 * Sets an object for the camera to follow.
 65 		 * @function
 66 		 * @name follow
 67 		 * @memberOf ex.display.AnimatedSprite
 68 		 * @param {Entity} object entity to follow
 69 		 */
 70 		follow: function (object) {
 71 			this.following = object;
 72 		}
 73 	});
 74 });
