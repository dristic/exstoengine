1 (function () {
  2 	ex.define("ex.world.Tile", {
  3 		
  4 		/**
  5 		 * An individial tile in a TileMap.
  6 		 * 
  7 		 * @name ex.world.Tile
  8 		 * 
  9 		 * @param {Number} value
 10 		 * @param {ex.base.Vector} position
 11 		 * @param {Number} width
 12 		 * @param {Number} height
 13 		 * @param {ex.world.Tile} neighborTop
 14 		 * @param {ex.world.Tile} neighborBottom
 15 		 * @param {ex.world.Tile} neighborLeft
 16 		 * @param {ex.world.Tile} neighborRight
 17 		 * 
 18 		 * @property {Number} value
 19 		 * @property {ex.base.Vector} position
 20 		 * @property {Number} width
 21 		 * @property {Number} height
 22 		 * @property {Number} mass used in collision detection
 23 		 * 
 24 		 * @constructor
 25 		 */
 26 		constructor: function(value, position, width, height, neighborTop, neighborBottom, neighborLeft, neighborRight) {
 27 			this.value = value;
 28 			this.position = position;
 29 			this.width = width;
 30 			this.height = height;
 31 			this.mass = 1;
 32 
 33 			/**
 34 			 * @name neighbors
 35 			 * @memberOf ex.world.Tile
 36 			 * 
 37 			 * @property {ex.world.Tile} top
 38 			 * @property {ex.world.Tile} bottom
 39 			 * @property {ex.world.Tile} left
 40 			 * @property {ex.world.Tile} right
 41 			 */
 42 			this.neighbors = {
 43 					top: 	neighborTop,
 44 					bottom: neighborBottom,
 45 					left: 	neighborLeft,
 46 					right: 	neighborRight
 47 			};
 48 			
 49 			/**
 50 			 * Used in collision edge detection. Each property is true 
 51 			 * if the neighbor on that side is null or valued as a
 52 			 * non-colliding tile (eg: empty space).
 53 			 * 
 54 			 * @name edges
 55 			 * @memberOf ex.world.Tile
 56 			 * 
 57 			 * @property {Boolean} top
 58 			 * @property {Boolean} bottom
 59 			 * @property {Boolean} left
 60 			 * @property {Boolean} right
 61 			 */
 62 			this.edges = {
 63 					top: 	false,
 64 					bottom: false,
 65 					left: 	false,
 66 					right: 	false
 67 			};
 68 			
 69 			this.update();
 70 		},
 71 		
 72 		/**
 73 		 * The update call. Generally only called when a tile's value
 74 		 * is changed or indirectly called when a neighboring tile's
 75 		 * value changes.
 76 		 * 
 77 		 * @function
 78 		 * @name update
 79 		 * @memberOf ex.world.Tile
 80 		 * 
 81 		 * @param {Boolean} requestedByNeighbor If true, this update request was 
 82 		 * 		called by a neighbor. This results in the tile not updating 
 83 		 * 		its own neighbors which would result in a resonance cascade 
 84 		 * 		a la Half-Life. We wouldn't want that now, would we?
 85 		 */
 86 		update: function(requestedByNeighbor) {
 87 			if(this.value == 0){
 88 				this._setAllEdgesTo(false);
 89 			}
 90 			// Set edge based on upper neighbor
 91 			if(!this.neighbors.top || this.neighbors.top.value == 0) {
 92 				this.edges.top = true;
 93 			} else {
 94 				this.edges.top = false;
 95 			}
 96 			// Set edge based on lower neighbor
 97 			if(!this.neighbors.bottom || this.neighbors.bottom.value == 0) {
 98 				this.edges.bottom = true;
 99 			} else {
100 				this.edges.bottom = false;
101 			}
102 			// Set edge based on left neighbor
103 			if(!this.neighbors.left || this.neighbors.left.value == 0) {
104 				this.edges.left = true;
105 			} else {
106 				this.edges.left = false;
107 			}
108 			// Set edge based on right neighbor
109 			if(!this.neighbors.right || this.neighbors.right.value == 0) {
110 				this.edges.right = true;
111 			} else {
112 				this.edges.right = false;
113 			}
114 			
115 			// Keeps the update request from cascading across the map
116 			if(!requestedByNeighbor){
117 				this.updateNeighbors();
118 			}
119 		},
120 		
121 		/**
122 		 * Calls each neighbor's update function if the neighbor exists.
123 		 * 
124 		 * @function
125 		 * @name updateNeighbors
126 		 * @memberOf ex.world.Tile
127 		 */
128 		updateNeighbors: function() {
129 			if(this.neighbors.up){
130 				this.neighbors.up.update(true);
131 			}
132 			if(this.neighbors.down){
133 				this.neighbors.down.update(true);
134 			}
135 			if(this.neighbors.left){
136 				this.neighbors.left.update(true);
137 			}
138 			if(this.neighbors.right){
139 				this.neighbors.right.update(true);
140 			}
141 		},
142 		
143 		/**
144 		 * Sets the value of the tile to the given value.
145 		 * 
146 		 * @function
147 		 * @name setValue
148 		 * @memberOf ex.world.Tile
149 		 * 
150 		 * @param {Number} value
151 		 */
152 		setValue: function(value) {
153 			this.value = value;
154 			this.update();
155 		},
156 		
157 		/**
158 		 * Sets the position of the tile to the given value.
159 		 * Not recommended for use, can cause lots of confusion with
160 		 * neighbors. I don't even know why I left this in here.
161 		 * 
162 		 * @function
163 		 * @name setPosition
164 		 * @memberOf ex.world.Tile
165 		 * 
166 		 * @param {Number} x
167 		 * @param {Number} y
168 		 */
169 		setPosition: function(x, y) {
170 			this.position.x = x;
171 			this.position.y = y;
172 		},
173 		
174 		_setAllEdgesTo: function(value) {
175 			this.edges.up 		= 	value;
176 			this.edges.down 	= 	value;
177 			this.edges.left 	= 	value;
178 			this.edges.right 	= 	value;
179 		}
180 	});
181 }());
