1 ex.using([
  2     'ex.util.CollisionDetector',
  3     'ex.util.CollisionResolver'
  4 ], function(){
  5 	ex.define("ex.util.CollisionManager", {
  6 		
  7 		/**
  8 		 * The engine component that handles collision detection and
  9 		 * resolution.
 10 		 * 
 11 		 * @name ex.util.CollisionManager
 12 		 * 
 13 		 * @property {ex.simplex.Map} activeLevel the current level in the game
 14 		 * 		world.
 15 		 * 
 16 		 * @constructor
 17 		 */
 18 		constructor: function() {
 19 			this.activeLevel = null;
 20 			this.detector = new ex.util.CollisionDetector();
 21 			this.resolver = new ex.util.CollisionResolver();
 22 			
 23 			this.benchmarkData = [];
 24 			this.benchmarkAverage = {
 25 					time: 0,
 26 					collisions: 0
 27 			};
 28 		},
 29 		
 30 		/**
 31 		 * Sets the level for the manager to handle collisions on.
 32 		 * 
 33 		 * @function
 34 		 * @name setActiveLevel
 35 		 * @memberOf ex.util.CollisionManager
 36 		 * 
 37 		 * @param {ex.simplex.Map} level
 38 		 */
 39 		setActiveLevel: function(level) {
 40 			this.activeLevel = level;
 41 			console.log("Active Level: " + this.activeLevel.name);
 42 		},
 43 		
 44 		updateBenchmark: function(newTime, newCollisionCount) {
 45 			if(this.benchmarkData.length > 200){
 46 				this.updateBenchmarkAverage();
 47 				this.benchmarkData = [];
 48 			}
 49 			this.benchmarkData.push({ 
 50 				time: newTime,
 51 				collisions: newCollisionCount
 52 			});
 53 		},
 54 		
 55 		updateBenchmarkAverage: function() {
 56 			var index = 0;
 57 			var sumTime = 0;
 58 			var sumCollisions = 0;
 59 			var dataCount = this.benchmarkData.length;
 60 			for(index; index < dataCount; index++){
 61 				sumTime += this.benchmarkData[index].time;
 62 				sumCollisions += this.benchmarkData[index].collisions;
 63 			}
 64 			this.benchmarkAverage.time = sumTime / dataCount;
 65 			this.benchmarkAverage.collisions = sumCollisions / dataCount;
 66 		},
 67 		
 68 		/**
 69 		 * Clears the collisions array, request collision detection on each
 70 		 * layer separately from the detector, then requests resolution of
 71 		 * each collision from the resolver.
 72 		 * 
 73 		 * @function
 74 		 * @name update
 75 		 * @memberOf ex.util.CollisionManager
 76 		 * 
 77 		 * @param {Number} dt timestep
 78 		 */
 79 		update: function(dt) {
 80 			ex.Debug.time('collision');
 81 			
 82 			// Do nothing if there is no active level
 83 			if(this.activeLevel == null){
 84 				return;
 85 			}
 86 			
 87 			var collisions = [];
 88 			// Loop through layers and detect collisions
 89 			var index = this.activeLevel.layers.length;
 90 			while(index--){
 91 				// Test spriteMap -> entities
 92 				collisions.push.apply(
 93 						collisions, 
 94 						this.detector.detectGroupCollisions(this.activeLevel.layers[index].items, dt));
 95 			}
 96 			this.resolver.resolveCollisions(collisions, dt);
 97 				
 98 			ex.Debug.time('collision');
 99 		}
100 	});	
101 });
