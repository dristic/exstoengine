1 ex.using([
  2     'ex.base.GlobalComponent',
  3     'ex.sound.Sound',
  4     'ex.event.EventTarget'
  5 ],function () {
  6 	ex.define("ex.util.AssetManager", ex.base.GlobalComponent, {
  7 		/**
  8 		 * The global asset manager, loads and retrieves audio, video, 
  9 		 * and image files.
 10 		 * 
 11 		 * @name ex.Assets
 12 		 * @class
 13 		 */
 14 		statics: {
 15 			componentName: 'Assets',
 16 			
 17 			_audio: [],
 18 			_video: [],
 19 			_images: [],
 20 			_ready: true,
 21 			_assetsToLoad: 0,
 22 			_assetsLoaded: 0,
 23 	
 24 			_supportedExtensions: {
 25 				video: [
 26 				    // MP4 (Chrome, IE9, Safari, iOS, Android)
 27 				    '.mp4', '.m4v', '.f4v', '.mov',
 28 				    // WebM (FF, Chrome, Opera)
 29 				    '.webm',
 30 				    // Vorbis (FF, Chrome, Opera)
 31 				    '.ogv'
 32 				    ],
 33 				audio: [
 34 				    // AAC (Chrome, IE9, Safari, iOS, Android)
 35 				    '.aac', '.m4a', '.f4a',
 36 				    // Vorbis (FF, Chrome, Opera, Android)
 37 				    '.ogg', '.oga',
 38 				    // MP3 (Chrome, IE9, Safari, iOS, Android)
 39 				    '.mp3'
 40 				    ],
 41 				image: [
 42 				    // Full Support
 43 				    '.jpg', '.jpeg', '.jpe', '.jif', '.jfif', '.jfi',
 44 				    '.png',
 45 				    '.gif',
 46 				    '.bmp', '.dib'
 47 				    ]
 48 			},
 49 			
 50 			/**
 51 			 * Retrieves an audio file by name from the asset manager.
 52 			 * 
 53 			 * @function
 54 			 * @name getAudio
 55 			 * @memberOf ex.Assets
 56 			 * 
 57 			 * @param {String} name
 58 			 * @returns {ex.sound.Sound} audio file or null with error
 59 			 */
 60 			getAudio: function(name) {
 61 				return this._audio[name] || 
 62 					throwAssetDoesNotExistError(name, 'audio');
 63 			},
 64 			
 65 			getVideo: function(name) {
 66 				return this._video[name] || 
 67 					throwAssetDoesNotExistError(name, 'video');
 68 			},
 69 			
 70 			/**
 71 			 * Retrieves an image file by name from the asset manager.
 72 			 * 
 73 			 * @function
 74 			 * @name getImage
 75 			 * @memberOf ex.Assets
 76 			 * 
 77 			 * @param {String} name
 78 			 * @returns {Image} image file or null with error
 79 			 */
 80 			getImage: function(name) {
 81 				if(!this._ready){
 82 					console.log("Cannot retrieve " + name + " image. Assets are still loading.");
 83 				}
 84 				return this._images[name] || 
 85 					throwAssetDoesNotExistError(name, 'image');
 86 			},
 87 			
 88 			/**
 89 			 * Loads a file from the file system and adds it to the
 90 			 * asset manager.
 91 			 * 
 92 			 * @function
 93 			 * @name load
 94 			 * @memberOf ex.Assets
 95 			 * 
 96 			 * @param {String} name name used to retrieve asset once loaded
 97 			 * @param {String} filePath path to the asset in the file system
 98 			 * @param {Object} [options] extra parameters, varies
 99 			 * 		by asset type.
100 			 */
101 			load: function (name, filePath, options) {
102 				// Determine file type and use proper loading method
103 				var extension = filePath.substring(filePath.lastIndexOf('.'));
104 				if(this._supportedExtensions.image.indexOf(extension) > -1) {
105 					this._loadImage(name, filePath, options);
106 				} else if (this._supportedExtensions.audio.indexOf(extension) > -1) {
107 					this._loadAudio(name, filePath, options);
108 				} else if (this._supportedExtensions.video.indexOf(extension) > -1) {
109 					this._loadVideo(name, filePath, options);
110 				} else {
111 					throwFileTypeNotSupportedError(name, filePath, extension);
112 				}
113 			},
114 			
115 			loadBulk: function(list){
116 				var index = list.length;
117 				while(index--){
118 					this.load(list[index].name, list[index].filePath, list[index].options);
119 				}
120 			},
121 			
122 			_loadImage: function(name, filePath, options) {
123 				if(this._images[name]){
124 					console.error(name + " cannot be loaded. This image already exists.");
125 					return;
126 				}
127 				this._images[name] = new Image();
128 				
129 				this._ready = false;
130 				this._assetsToLoad++;
131 				
132 				var that = this;
133 				this._images[name].onError = throwFileUnableToLoadError;
134 				this._images[name].onload = function () {
135 					that._assetsLoaded++;
136 					if(that._assetsLoaded == that._assetsToLoad) {
137 						that._ready = true;
138 					}
139 				};
140 				
141 				this._images[name].src = '';
142 				this._images[name].src = filePath;
143 			},
144 			
145 			_loadAudio: function(name, filePath, options) {
146 				var numChannels = 4;
147 				if(options && options.numChannels){
148 					numChannels = options.numChannels;
149 				}
150 				
151 				var exSound = new ex.sound.Sound(new Audio()),
152 					that = exSound;
153 				
154 				this._ready = false;
155 				this._assetsToLoad++;
156 				
157 				that.audio.onError = throwFileUnableToLoadError;
158 				that.audio.src = filePath;
159 				that.audio.addEventListener('canplaythrough', function (event) {
160 					while(numChannels--) {
161 						that.channels.push(that.audio.cloneNode(true));
162 						that.readyChannels.push(true);
163 						that.ready = true;
164 					}
165 				});
166 				
167 				this._audio[name] = exSound;
168 			},
169 			
170 			_loadVideo: function(name, filePath, options) {
171 				console.error("No support for video loading yet. Maybe next time!");
172 			}
173 		}
174 	});
175 	
176 	function throwAssetDoesNotExistError(name, type) {
177 		console.error("The " + type + " file '" + name + "' does not exist. Maybe you forgot to load it.");
178 	};
179 	
180 	function throwFileUnableToLoadError(filePath) {
181 		console.error("An error occured while loading the file at '" + filePath + "'.");
182 	};
183 	
184 	function throwFileTypeNotSupportedError(name, filePath, extension) {
185 		console.error(
186 				'Not loading  "' + name + 
187 				'" from "' + filePath + 
188 				'" because the extension "' + extension + 
189 				'" is not supported.');
190 	}
191 	
192 	ex.Assets = ex.util.AssetManager;
193 });
