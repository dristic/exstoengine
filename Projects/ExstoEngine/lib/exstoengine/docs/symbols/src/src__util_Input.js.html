1 ex.using([
  2   "ex.util.Key"
  3 ], function () {
  4 	ex.define("ex.util.Input", {
  5 		constructor: function () {
  6 			this.keys = [];
  7 			this.pressed = [];
  8 			this.pressedTimer = 1;
  9 			this.mouseX = 0;
 10 			this.mouseY = 0;
 11 			this.mouseDown = false;
 12 			this.mouseUp = false;
 13 			this.rightButton = false;
 14 			
 15 			this.lastMouseX = 0;
 16 			this.lastMouseY = 0;
 17 			this.dragging = false;						// If the mouse is dragging or not
 18 			this._beginDrag = false;					// Buffer so dragging does not always happen instantly
 19 		},
 20 		
 21 		listenOn: function(element) {
 22 			document.addEventListener("keydown", ex.bind(this, this.onKeyDown));
 23 			document.addEventListener("keyup", ex.bind(this, this.onKeyUp));
 24 			element.addEventListener("mousedown", ex.bind(this, this.onMouseDown));
 25 			element.addEventListener("mouseup", ex.bind(this, this.onMouseUp));
 26 			element.addEventListener("mousemove", ex.bind(this, this.onMouseMove));
 27 		},
 28 		
 29 		update: function(dt) {			
 30 			var i = this.pressed.length;
 31 			while(i--) {
 32 				if(this.pressed[i] > 0)
 33 				this.pressed[i]--;
 34 			}
 35 			
 36 			if(this._beginDrag == true && this.mouseUp == false) {
 37 				this.dragging = true;
 38 			}
 39 			
 40 			this.mouseDown = false;
 41 			this.mouseUp = false;
 42 			
 43 			this.lastMouseX = this.mouseX;
 44 			this.lastMouseY = this.mouseY;
 45 		},
 46 		
 47 		getMousePosition: function(event) {
 48 			// Firefox, IE9, & Chrome
 49 			if(event.layerX || event.layerX == 0) {
 50 				event._x = event.layerX;
 51 				event._y = event.layerY;
 52 			}
 53 			// Opera
 54 			else if(event.offsetX || event.offsetX == 0) {
 55 				event._x = event.offsetX;
 56 				event._y = event.offsetY;
 57 			}
 58 			// Unsupported
 59 			else {
 60 				event._x = -1;
 61 				event._y = -1;
 62 			}
 63 			
 64 			return event;
 65 		},
 66 		
 67 		getMouseDelta: function () {
 68 			return [(this.mouseX - this.lastMouseX), (this.mouseY - this.lastMouseY)];
 69 		},
 70 		
 71 		onMouseDown: function(event) {
 72 			this.mouseDown = true;
 73 			this._beginDrag = true;
 74 			
 75 			if (event.which) this.rightButton = (event.which == 3);
 76 			else if (event.button) this.rightButton = (event.button == 2);
 77 		},
 78 		
 79 		onMouseUp: function(event) {
 80 			this.mouseUp = true;
 81 			this._beginDrag = false;
 82 			this.dragging = false;
 83 			this.rightButton = false;
 84 		},
 85 		
 86 		onMouseMove: function(event) {
 87 			event = this.getMousePosition(event);
 88 			this.mouseX = event._x;
 89 			this.mouseY = event._y;
 90 		},
 91 		
 92 		onKeyDown: function(event) {
 93 			if(this.keys[event.keyCode] == false || !this.keys[event.keyCode]) {
 94 				this.keys[event.keyCode] = true;
 95 				this.pressed[event.keyCode] = this.pressedTimer;	
 96 			}
 97 		},
 98 		
 99 		onKeyUp: function(event) {
100 			this.keys[event.keyCode] = false;
101 		},
102 		
103 		isKeyDown: function(keyCode) {
104 			return this.keys[keyCode];
105 		},
106 		
107 		isKeyPressed: function(keyCode) {
108 			if(this.pressed[keyCode] > 0) {
109 				return true;
110 			} else {
111 				return false;
112 			}
113 		}
114 	});
115 });
