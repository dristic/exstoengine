1 ex.using([
  2   "ex.base.Vector"
  3 ], function () {
  4 	ex.define("ex.display.Sprite", {
  5 		/**
  6 		 * Sprite object that contains a single image to draw on the canvas.
  7 		 * @name ex.display.Sprite
  8 		 * 
  9 		 * @param {Number} x The sprite's x position.
 10 		 * @param {Number} y The sprite's y position.
 11 		 * @param {Image} img The img to use to render the sprite.
 12 		 * @param {String} name The sprite's name.
 13 		 * @constructor
 14 		 */
 15         constructor: function (x, y, img, name) {
 16         	this.name = name;
 17         	this.type = "Sprite";
 18             this.x = x || 0;
 19             this.y = y || 0;
 20             this.position = new ex.base.Vector(x, y);
 21             this.velocity = new ex.base.Vector(0, 0);
 22             this.img = img || new Image();
 23             this.visible = true;
 24             this.width = this.img.width;
 25             this.height = this.img.height;
 26 
 27             this.halfWidth = function () {
 28                 return this.img.width >> 1;
 29             };
 30 
 31             this.halfHeight = function () {
 32                 return this.img.height >> 1;
 33             };
 34 
 35             this.rotation = 0;
 36             this.rotationEnabled = false;
 37             this.rotationCanvas = document.createElement("canvas");
 38 
 39             this.scrollFactorX = 1;
 40             this.scrollFactorY = 1;
 41 
 42             //--Set rotation canvas the size of the image
 43             this.rotationCanvas.width = this.img.width;
 44             this.rotationCanvas.height = this.img.height;
 45         },
 46 
 47         update: function (dt) {
 48             if (this.width == 0) {
 49                 this.width = this.img.width;
 50                 this.height = this.img.height;
 51             }
 52 
 53             if (typeof this.onUpdate === "function") this.onUpdate(dt);
 54         },
 55 
 56         render: function (context, camX, camY) {
 57             if (!this.visible){
 58             	return;
 59             }
 60             
 61             if (this.rotationEnabled == false) {
 62                 context.drawImage(this.img, this.x - (camX * this.scrollFactorX), this.y - (camY * this.scrollFactorY));
 63             } else {
 64                 var rContext = this.rotationCanvas.getContext("2d");
 65 
 66                 //--Ensure width and height are not 0 to avoid INVALID_STATE_ERR
 67                 this.rotationCanvas.width = this.img.width || 1;
 68                 this.rotationCanvas.height = this.img.height || 1;
 69 
 70                 rContext.save();
 71                 rContext.translate(this.halfWidth(), this.halfHeight());
 72                 rContext.rotate(this.rotation);
 73                 rContext.translate(-this.halfWidth(), -this.halfHeight());
 74                 rContext.drawImage(this.img, 0, 0);
 75                 rContext.restore();
 76 
 77                 context.drawImage(this.rotationCanvas, this.x - (camX * this.scrollFactorX), this.y - (camY * this.scrollFactorY));
 78             }
 79         },
 80 
 81         handleCollision: function (collision) {
 82             if (collision.y == true) {
 83                 this.velocity.y = 0;
 84                 this.position.y = collision.position.y;
 85             }
 86             if (collision.x == true) {
 87                 this.velocity.x = 0;
 88                 this.position.x = collision.position.x;
 89             }
 90         },
 91 
 92         getBounds: function () {
 93             return new ExstoEngine.Base.Rectangle(this.x, this.x, this.width, this.height);
 94         }
 95     });
 96 
 97 });
