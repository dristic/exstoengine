1 ex.using([
  2   "ex.base.Vector"
  3 ], function () {
  4 	ex.define("ex.display.Sprite", {
  5 		/**
  6 		 * Sprite object that contains a single image to draw on the canvas.
  7 		 * 
  8 		 * @name ex.display.Sprite
  9 		 * 
 10 		 * @param {Number} x The sprite's x position.
 11 		 * @param {Number} y The sprite's y position.
 12 		 * @param {Image} img The img to use to render the sprite.
 13 		 * @param {String} name The sprite's name.
 14 		 * 
 15 		 * @property {String} type
 16 		 * 		Internal use only. Do not change!
 17 		 * @property {ex.base.Vector} position
 18 		 * @property {ex.display.Image} image
 19 		 * @property {Number} rotation angle of the image
 20 		 * @property {Boolean} rotationEnabled if rotation is currently
 21 		 * 		being used.
 22 		 * @property {Canvas} rotationCanvas buffer canvas used to rotate
 23 		 * 		the sprite.
 24 		 * @property {ex.base.Vector} scrollFactor the rate at which the
 25 		 * 		sprite moves with the camera. Values less than 1 cause the
 26 		 * 		sprite to move slower than the camera making it appear to be
 27 		 * 		farther away from the focus while numbers greater than 1 
 28 		 *		make the sprite appear to be closer than the focus.
 29 		 * @constructor
 30 		 */
 31         constructor: function (position, img) {
 32         	this.type = "Sprite";
 33             this.position = position;
 34             this.img = img || new Image();
 35             this.visible = true;
 36 
 37             this.rotation = 0;
 38             this.rotationEnabled = false;
 39             this.rotationCanvas = document.createElement("canvas");
 40 
 41             this.scrollFactor = new ex.base.Vector(1,1);
 42             
 43             this.width = this.img.naturalWidth;
 44             this.height = this.img.naturalHeight;
 45             this.rotationCanvas.width = this.width;
 46             this.rotationCanvas.height = this.height;
 47             
 48             if(this.width == 0  && this.height == 0) {
 49             	ex.event.listen(img, 'load', function () {
 50             		this._recalcDimensions();
 51             	}, this);
 52             }
 53         },
 54         
 55         /**
 56          * Recalculates dimensions of sprite based on image. 
 57          * Automatically called if image changes or loads after sprite 
 58          * is initialized.
 59          * 
 60          * @function
 61          * @name _recalcDimensions
 62          * @memberOf ex.display.Sprite
 63          */
 64         _recalcDimensions: function () {
 65         	this.width = this.img.naturalWidth;
 66             this.height = this.img.naturalHeight;
 67             this.rotationCanvas.width = this.width;
 68             this.rotationCanvas.height = this.height;
 69         },
 70 
 71         /**
 72          * Update routine
 73          * 
 74          * @function
 75          * @name update
 76          * @memberOf ex.display.Sprite
 77          * 
 78          * @param {Number} dt timestep
 79          */
 80         update: function (dt) {
 81             if (typeof this.onUpdate === "function") this.onUpdate(dt);
 82         },
 83         
 84         /**
 85          * Renders sprite, usually called by Renderer.
 86          * 
 87          * @function
 88          * @name render
 89          * @memberOf ex.display.Sprite
 90          * 
 91          * @param {Context} context
 92          * @param {Number} camX
 93          * @param {Number} camY
 94          */
 95         render: function (context, camX, camY) {
 96             if (!this.visible){
 97             	return;
 98             }
 99             
100             if (this.rotationEnabled == false) {
101                 context.drawImage(
102                 		this.img, 
103                 		this.position.x - (camX * this.scrollFactor.x), 
104                 		this.position.y - (camY * this.scrollFactor.y));
105             } else {
106                 var rContext = this.rotationCanvas.getContext("2d");
107 
108                 //--Ensure width and height are not 0 to avoid INVALID_STATE_ERR
109                 this.rotationCanvas.width = this.img.width || 1;
110                 this.rotationCanvas.height = this.img.height || 1;
111 
112                 rContext.save();
113                 rContext.translate(this.width / 2, this.height / 2);
114                 rContext.rotate(this.rotation);
115                 rContext.translate(-this.width / 2, -this.height / 2);
116                 rContext.drawImage(this.img, 0, 0);
117                 rContext.restore();
118 
119                 context.drawImage(
120                 		this.rotationCanvas, 
121                 		this.position.x - (camX * this.scrollFactor.x), 
122                 		this.position.y - (camY * this.scrollFactor.y));
123             }
124         },
125 
126         /**
127          * Returns bounding box of sprite.
128          * 
129          * @function
130          * @name getBounds
131          * @memberOf ex.display.Sprite
132          * @returns {ex.base.Rectangle} bounding box
133          */
134         getBounds: function () {
135             return new ex.base.Rectangle(
136             		this.position, 
137             		this.width, 
138             		this.height);
139         }
140     });
141 
142 });
