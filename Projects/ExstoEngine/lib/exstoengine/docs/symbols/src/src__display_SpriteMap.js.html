1 ex.using([
  2   "ex.world.TileMap",
  3   //"ex.base.Vector"
  4 ], function () {
  5 	ex.define("ex.display.SpriteMap", ex.world.TileMap, {
  6 		
  7 		/**
  8 		 * The graphical representation of a TileMap.
  9 		 * 
 10 		 * @name ex.display.SpriteMap
 11 		 * @extends ex.world.TileMap
 12 		 * 
 13 		 * @param {Number} tileWidth
 14 		 * @param {Number} tileHeight
 15 		 * @param {Number[]} map a 2D array of integer values relating to the
 16 		 * 		tiles in the supplied tileSet
 17 		 * @param {Image} tileSet
 18 		 * @param {String} name
 19 		 * 
 20 		 * @property {Boolean} visible Controls whether the object is drawn
 21 		 * 		to the screen.
 22 		 * @property {Boolean} collides Controls whether the object is included
 23 		 * 		in collision detection.
 24 		 * @propety {String} name
 25 		 * @property {String} type
 26 		 * 		Internal use only. Do not change!
 27 		 * @property {Image} tileSet the tile set to use when rendering.
 28 		 * @property {ex.base.Vector} position
 29 		 * @property {ex.base.Vector} scrollFactor
 30 		 */
 31 		constructor: function(tileWidth, tileHeight, map, tileSet, name) {
 32 			this.visible = true;
 33 			this.collides = false;
 34 			this.name = name;
 35 			this.type = "SpriteMap";
 36 			this.tileSet = tileSet;
 37 			this.position = new ex.base.Vector(0,0);
 38 			this.scrollFactor = new ex.base.Vector(1,1);
 39 			
 40 			this._super("constructor", [tileWidth, tileHeight, map]);
 41 		},
 42 		
 43 		/**
 44 		 * Contains any event handling when a collision occurs between this
 45 		 * and another object.
 46 		 * 
 47 		 * @function
 48 		 * @name onCollide
 49 		 * @memberOf ex.display.SpriteMap
 50 		 */
 51 		onCollide: function() {
 52 			
 53 		},
 54 		
 55 		/**
 56 		 * The update routine.
 57 		 * 
 58 		 * @function
 59 		 * @name update
 60 		 * @memberOf ex.display.SpriteMap
 61 		 * 
 62 		 * @param {Number} dt timestep
 63 		 */
 64 		update: function(dt) {
 65 			
 66 		},
 67 		
 68 		/**
 69 		 * The render routine.
 70 		 * 
 71 		 * @function
 72 		 * @name render
 73 		 * @memberOf ex.display.SpriteMap
 74 		 * 
 75 		 * @param {Context} context
 76 		 * @param {Number} camX
 77 		 * @param {Number} camY
 78 		 * @param {Number} camWidth
 79 		 * @param {Number} camHeight
 80 		 */
 81 		render: function(context, camX, camY, camWidth, camHeight) {
 82 			// Do nothing if not visible
 83 			if(!this.visible){
 84 				return;
 85 			}
 86 			
 87 			//Pre-calculated values for speed
 88 			var camWithScrollX = (camX * this.scrollFactor.x),
 89 				camWithScrollY = (camY * this.scrollFactor.y);
 90 			
 91 			// Calculate the range of tiles in the viewport
 92 			var yStart = ex.toInt(camWithScrollY / this.tileHeight),
 93 				yStop = ex.toInt((camWithScrollY + camHeight) / this.tileHeight + 1),
 94 				xStart = ex.toInt(camWithScrollX / this.tileWidth),
 95 				xStop = ex.toInt((camWithScrollX + camWidth) / this.tileWidth + 1);
 96 			
 97 			// Check for the SpriteMap being completely off screen
 98 			if(yStart < ex.toInt(-(camHeight / this.tileHeight)))
 99 				return;		// SpriteMap below viewport
100 			else if(yStop < 0)
101 				return;		// SpriteMap above viewport
102 			else if(xStart < ex.toInt(-(camWidth / this.tileWidth)))
103 				return;		// SpriteMap left of viewport
104 			else if(xStop < 0)
105 				return;		// SpriteMap right of viewport
106 			
107 			// Constrain start/stop parameters to grid values
108 			if(yStart < 0)
109 				yStart = 0;
110 			else if(yStart > this.data.length)
111 				yStop = this.data.length;
112 			if(xStart < 0)
113 				xStart = 0;
114 			else if(xStart > this.data[0].length)
115 				xStart = this.data[0].length;
116 			if(yStop > this.data.length)
117 				yStop = this.data.length;
118 			if(xStop > this.data[0].length)
119 				xStop = this.data[0].length;
120 			
121 			// Render tiles in viewport
122 			var yPos = yStart,
123 				xPos = xStart;
124 			for(yPos; yPos < yStop; yPos++) {
125 				for(xPos; xPos < xStop; xPos++) {
126 					var tile = this.data[yPos][xPos], sx = 0, sy = 0;
127 					var tileValue = tile.value;
128 					if(tileValue != 0) {
129 						while(--tileValue) {
130 							sx += this.tileWidth;
131 							if(sx >= this.tileSet.width) {
132 								sy += this.tileHeight;
133 								sx = 0;
134 							}
135 						}
136 						context.drawImage(this.tileSet,
137 								      sx,
138 								      sy,
139 								      this.tileWidth,
140 								      this.tileHeight,
141 								      ex.toInt(tile.position.x - camWithScrollX),
142 								      ex.toInt(tile.position.y - camWithScrollY),
143 								      tile.width,
144 								      tile.height);
145 					}
146 				}
147 				xPos = xStart;
148 			}
149 		}
150 	});
151 });
