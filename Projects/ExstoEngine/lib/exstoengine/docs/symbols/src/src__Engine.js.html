1 ex.using([
  2           "ex.base.Component",
  3           "ex.base.Point",
  4           "ex.util.Input",
  5           "ex.util.Debug",
  6           "ex.util.AssetManager",
  7           "ex.display.ImageRepository",
  8           "ex.util.CollisionManager",
  9           "ex.display.Camera",
 10           "ex.display.Renderer"
 11           ],
 12 	function () {
 13 	
 14 	ex.define("ex.Engine", {
 15 		/**
 16 		 * The base engine class for loading games and drawing to the canvas
 17 		 * @name ex.Engine
 18 		 * 
 19 		 * @param {Int} width The width of the canvas
 20 		 * @param {Int} height The height of the canvas
 21 		 * @param {Int} frameRate The frame rate of the game
 22 		 * @constructor
 23 		 */
 24 		constructor: function (width, height, frameRate) {
 25 			//--Check for canvas support
 26 			if(document.createElement("canvas").getContext) {
 27 				//--Public
 28 				this.frameRate = frameRate || 50;
 29 				this.width = width;
 30 				this.height = height;
 31 				this.currentWorld = null;
 32 				this.lastTime = (new Date()).getTime();
 33 				this.components = [];
 34 				this.debug = false;
 35 				
 36 				var _gameInterval = null;
 37 				
 38 				//--Load up input class
 39 				this.input = new ex.util.Input();
 40 				
 41 				//--Load new image repository
 42 				this.imageRepository = new ex.display.ImageRepository();
 43 				
 44 				this.collisionManager = new ex.util.CollisionManager();
 45 				
 46 				//--Load new camera
 47 				this.camera = new ex.display.Camera(
 48 						new ex.base.Point(0,0),
 49 						this.width,
 50 						this.height);
 51 				
 52 				//--Setup update interval
 53 				_gameInterval = setInterval(ex.bind(this, this.update), (1 / frameRate) * 1000);
 54 			} else {
 55 				ex.Debug.log("Your browser does not support canvas!");
 56 			}
 57 		},
 58 		
 59 		enableDebugging: function() {
 60 			this.debug = true;
 61 			
 62 			ex.Debug.enable();
 63 		},
 64 		
 65 		setupCanvas: function (bgColor, canvas) {
 66 			this.renderer = new ex.display.Renderer(this.width, this.height, bgColor, canvas);
 67 			
 68 			this.camera.canvas = canvas || this.renderer.canvas;
 69 		},
 70 		
 71 		update: function () {
 72 			//--Calculate delta time
 73 			var newTime = (new Date()).getTime();
 74 			var dt = newTime - this.lastTime;
 75 			dt /= 1000;
 76 			this.lastTime = newTime;
 77 			
 78 			//--Step components
 79 			var i = this.components.length;
 80 			while(i--) {
 81 				this.components[i].update(dt);
 82 			}
 83 			
 84 			//--Step world
 85 			if(this.currentWorld != null) {
 86 				this.currentWorld.update(dt);
 87 			}
 88 			
 89 			//--Step collision manager
 90 			if(this.collisionManager != null) {
 91 				this.collisionManager.update(dt);
 92 			}
 93 			
 94 			//--Step camera
 95 			this.camera.update(dt);
 96 			
 97 			//--Step renderer
 98 			if(this.renderer != null) {
 99 				this.renderer.update(dt, this.camera);
100 			}
101 			
102 			this.onUpdate(dt);
103 			
104 			this.input.update(dt);
105 			
106 			ex.Debug.benchmarkEngine(dt);
107 		},
108 		
109 		onUpdate: function() {
110 			
111 		},
112 		
113 		openWorld: function(world) {
114 			if(this.currentWorld != null) {
115 				this.currentWorld.destroy();
116 			}
117 			
118 			this.currentWorld = new world(this.renderer);
119 		},
120 		
121 		loadLevel: function(levelName) {
122 			var levelNamespace = "game.levels." + levelName;
123 			
124 			ex.using([levelNamespace], function(){
125 				var level = new game.levels[levelName](this.input);
126 				var assets = level.getAssets();
127 
128 				ex.event.listen(ex.Assets, 'ready', function(){
129 					var objects = level.getObjects();
130 					this.currentWorld.addObjects(objects);
131 				}, this);
132 				
133 				ex.Assets.loadBulk(assets);
134 				
135 				
136 			});
137 			
138 		},
139 		
140 		loadComponent: function(component) {
141 			if(component instanceof ex.base.Component == false) {
142 				this.logger.log("Component must be an instance of ExstoEngine.Base.Component!");
143 			} else {
144 				this.components.push(component);
145 			}
146 		},
147 		
148 		getComponent: function(name) {
149 			var i = this.components.length;
150 			while(i--) {
151 				if(this.components[i].name == name) {
152 					return this.components[i];
153 				}
154 			}
155 		}
156 	});
157 });
