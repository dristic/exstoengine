1 ex.using([
  2     "ex.util.CollisionManager"
  3 ], function() {
  4 	ex.define("ex.world.World", {
  5 		
  6 		/**
  7 		 * The world object cotains all the relevant game data such as
  8 		 * levels, menus, and entities.
  9 		 * 
 10 		 * @name ex.world.World
 11 		 * 
 12 		 * @param {ex.display.Renderer} renderer the renderer to call
 13 		 * 		to render each frame
 14 		 * 
 15 		 * @property {ex.simplex.Map} activeLevel the current level
 16 		 * @proeprty {ex.display.Renderer} renderer
 17 		 * @property {ex.simplex.Map[]} levels all levels in the world
 18 		 * @property {Object[]} objects all global data such as player
 19 		 * 		stats, game stats, and menus.
 20 		 * 
 21 		 * @constructor
 22 		 */
 23 		constructor: function(renderer) {
 24 			this.activeLevel = null;
 25 			this.renderer = renderer;
 26 			this.levels = [];
 27 			this.objects = [];
 28 		},
 29 		
 30 		/**
 31 		 * Updates all levels and objects.
 32 		 * 
 33 		 * @function
 34 		 * @name update
 35 		 * @memberOf ex.world.World
 36 		 * 
 37 		 * @param {Number} dt timestep
 38 		 */
 39 		update: function(dt) {
 40 			// update level
 41 			if(this.activeLevel)
 42 				this.activeLevel.update(dt);
 43 			
 44 			// update objects
 45 			var i = this.objects.length;
 46 			while(i--) {
 47 				this.objects[i].update(dt);
 48 			}
 49 		},
 50 
 51 		/**
 52 		 * Retrieves a level by name.
 53 		 * 
 54 		 * @function
 55 		 * @name getLevel
 56 		 * @memberOf ex.world.World
 57 		 * 
 58 		 * @param {String} name
 59 		 * 
 60 		 * @returns {ex.simplex.Map}
 61 		 */
 62 		getLevel: function(name){
 63 			var index = this.levels.length;
 64 			while(index--){
 65 				if(this.levels[index].name == name){
 66 					return this.levels[index];
 67 				}
 68 			}
 69 			
 70 			//if not found
 71 			return null;
 72 		},
 73 		
 74 		/**
 75 		 * Sets the active level to the level with the given name.
 76 		 * 
 77 		 * @function
 78 		 * @name loadLevel
 79 		 * @memberOf ex.world.World
 80 		 * 
 81 		 * @param {String} name
 82 		 */
 83 		loadLevel: function(name){
 84 			this._setActiveLevel(this.getLevel(name));
 85 			console.log("World activeLevel: " + this.activeLevel.name);
 86 		},
 87 		
 88 		_setActiveLevel: function(level){
 89 			if(this.activeLevel != null){
 90 				// unload renderables
 91 				var index = this.renderer.renderables.length;
 92 				while(index--){
 93 					if(this.renderer.renderables[index] === this.activeLevel){
 94 						this.renderer.renderables.splice(index, 1);
 95 					}	
 96 				}
 97 			}
 98 			
 99 			// set activeLevel
100 			this.activeLevel = level;
101 			
102 			// add renderables
103 			this.renderer.renderables.push(level);
104 		},
105 		
106 		/**
107 		 * Adds a level to the world.
108 		 * 
109 		 * @function
110 		 * @name addLevel
111 		 * @memberOf ex.world.World
112 		 * 
113 		 * @param {ex.simplex.Map} level
114 		 */
115 		addLevel: function(level) {
116 			this.levels.push(level);
117 		},
118 		
119 		/**
120 		 * Adds an object to the world by direct reference.
121 		 * 
122 		 * @function
123 		 * @name addObject
124 		 * @memberOf ex.world.World
125 		 * 
126 		 * @param {Object} object
127 		 */
128 		addObject: function(object) {
129 			this.objects.push(object);
130 			this.renderer.renderables.push(object);
131 		},
132 		
133 		addObjects: function(objects) {
134 			var index = objects.length;
135 			while(index--){
136 				this.addObject(objects[index]);
137 			}
138 		},
139 		
140 		/**
141 		 * Removes an object from the world by direct reference.
142 		 * 
143 		 * @function
144 		 * @name removeObject
145 		 * @memberOf ex.world.World
146 		 * 
147 		 * @param {Object} object
148 		 */
149 		removeObject: function(object) {
150 			// Remove object from world
151 			var index = this.objects.length;
152 			while(index--) {
153 				if(this.objects[index] === object){
154 					this.objects.splice(index, 1);
155 				}
156 			}
157 			
158 			// Remove object from renderer
159 			index = this.renderer.renderables.length;
160 			while(index--) {
161 				if(this.renderer.renderables[index] === object) {
162 					this.renderer.renderables.splice(index, 1);
163 				}
164 			}
165 		},
166 		
167 		destroy: function() {
168 			
169 		}
170 	});
171 });
